021    package xbird.engine.backend;
022    
023    import java.rmi.RemoteException;
024    import java.util.Arrays;
025    import java.util.Map;
026    
027    import org.apache.commons.logging.Log;
028    import org.apache.commons.logging.LogFactory;
029    
030    import xbird.client.command.*;
031    import xbird.engine.*;
032    import xbird.engine.Request.Signature;
033    import xbird.engine.request.CommandRequest;
034    import xbird.storage.DbCollection;
035    import xbird.util.concurrent.collections.ConcurrentIdentityHashMap;
036    
037    /**
038     * 
039     * <DIV lang="en"></DIV>
040     * <DIV lang="ja"></DIV>
041     * 
042     * @author Makoto YUI (yuin405+xbird@gmail.com)
043     */
044    public final class CommandProcessor extends BackendProcessor {
045        private static final Log LOG = LogFactory.getLog(CommandProcessor.class);
046    
047        private final Map<RequestContext, Thread> _runningThreads = new ConcurrentIdentityHashMap<RequestContext, Thread>(12);
048    
049        public CommandProcessor(ResponseListener handler) {
050            super(handler);
051        }
052    
053        public Signature associatedWith() {
054            return Signature.COMMAND;
055        }
056    
057        public void fire(RequestContext rc) {
058            rc.setFired(System.currentTimeMillis());
059    
060            Request request = rc.getRequest();
061            Signature rsig = request.getSignature();
062            if(rsig != Signature.COMMAND) {
063                throw new IllegalStateException("Illegal command is passed to CommandProcessor: "
064                        + rsig);
065            }
066            CommandRequest command = (CommandRequest) request;
067            String[] cmdArg = command.getArgs();
068            String baseCol = command.getBaseCollection();
069    
070            DbCollection contextCol = DbCollection.getCollection(baseCol);
071            Session session = new Session(contextCol);
072            CommandInvoker invoker = new CommandInvoker(session);
073    
074            _runningThreads.put(rc, Thread.currentThread());
075            try {
076                boolean status = invoker.executeCommand(cmdArg);
077                rc.setResult(status);
078            } catch (CommandException ce) {
079                LOG.error("command failed: " + Arrays.toString(cmdArg), ce);
080                rc.setFault(ce);
081            } finally {
082                _runningThreads.remove(rc);
083            }
084    
085            try {
086                _resHandler.onResponse(rc);
087            } catch (RemoteException re) {
088                LOG.error("The respond for the request '" + rc.getRequest().getIdentifier()
089                        + "' is failed", re);
090            }
091        }
092    
093        public void cancel(RequestContext rc) {
094            Thread thread = _runningThreads.remove(rc);
095            if(thread != null) {
096                thread.interrupt();
097                try {
098                    thread.join();
099                } catch (InterruptedException e) {
100                    Thread.currentThread().interrupt();
101                }
102            }
103        }
104    
105    }
