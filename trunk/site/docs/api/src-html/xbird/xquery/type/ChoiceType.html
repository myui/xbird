021    package xbird.xquery.type;
022    
023    import java.util.ArrayList;
024    import java.util.List;
025    
026    /**
027     * A choice of item types, called a prime type.
028     * <DIV lang="en"></DIV>
029     * <DIV lang="ja"></DIV>
030     * 
031     * @author Makoto YUI (yuin405+xbird@gmail.com)
032     * @link http://www.w3.org/TR/xquery-semantics/#jd_prime
033     */
034    public class ChoiceType extends Type {
035        private static final long serialVersionUID = 7751599145163518572L;
036        private final List<Type> _types;
037    
038        public ChoiceType(List<Type> t) {
039            this._types = t;
040        }
041    
042        public ChoiceType(Type... type) {
043            this(new ArrayList<Type>(type.length));
044            combine(type);
045        }
046    
047        public List<Type> combine(Type... type) {
048            for(Type t : type) {
049                if(!_types.contains(type)) {
050                    _types.add(t);
051                }
052            }
053            return this._types;
054        }
055    
056        public boolean accepts(Type expected) {
057            final Type prime = expected.prime();
058            if(prime == null) { // for SequenceType$Empty
059                return false;
060            }
061            final List<Type> choice = ((ChoiceType) prime).getTypes();
062            for(Type t : _types) {
063                if(prime instanceof ChoiceType) {
064                    for(Type c : choice) {
065                        if(t.accepts(c)) {
066                            return true;
067                        }
068                    }
069                } else {
070                    if(t.accepts(prime)) {
071                        return true;
072                    }
073                }
074            }
075            return false;
076        }
077    
078        public Class getJavaObjectType() {
079            throw new UnsupportedOperationException("getJavaObjectType() for PrimeType is not supported.");
080        }
081    
082        public Occurrence quantifier() {
083            return _types.isEmpty() ? Occurrence.OCC_ZERO_OR_MORE : Occurrence.OCC_ONE_OR_MORE;
084        }
085    
086        public Type prime() {
087            return this;
088        }
089    
090        public List<Type> getTypes() {
091            return _types;
092        }
093    
094        public String toString() {
095            final StringBuilder buf = new StringBuilder(64);
096            buf.append('(');
097            for(int i = 0; i < _types.size(); i++) {
098                if(i != 0) {
099                    buf.append(" | ");
100                }
101                Type t = _types.get(i);
102                buf.append(t.toString());
103            }
104            buf.append(')');
105            return buf.toString();
106        }
107    
108    }
