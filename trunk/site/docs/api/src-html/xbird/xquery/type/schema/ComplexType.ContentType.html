032    public class ComplexType extends Type implements SchemaType {
033        private static final long serialVersionUID = -5786783608331802325L;
034    
035        public enum ContentType {
036            ANY((byte) 127), MixedContent((byte) 1), SimpleContent((byte) 2), EmptyContent((byte) 4);
037            private final byte bit;
038    
039            ContentType(byte b) {
040                bit = b;
041            }
042    
043            public boolean isType(ContentType type) {
044                final byte mask = type.getValue();
045                return (bit & mask) == mask;
046            }
047    
048            public byte getValue() {
049                return bit;
050            }
051        }
052    
053        protected final ContentType _contentType;
054    
055        public ComplexType() {
056            this(ContentType.ANY);
057        }
058    
059        public ComplexType(ContentType type) {
060            super();
061            this._contentType = type;
062        }
063    
064        public boolean accepts(Type expected) {
065            return false;
066        }
067    
068        public Class getJavaObjectType() {
069            return null;
070        }
071    
072        public Occurrence quantifier() {
073            return Occurrence.OCC_ZERO_OR_MORE;
074        }
075    
076        public ContentType getContentType() {
077            return _contentType;
078        }
079    
080        public Type prime() {
081            return this; // FIXME
082        }
083    
084        public String toString() {
085            throw new UnsupportedOperationException(ComplexType.class.getSimpleName()
086                    + "#toString() is not implementated."); // FIXME
087        }
088    
089    }
