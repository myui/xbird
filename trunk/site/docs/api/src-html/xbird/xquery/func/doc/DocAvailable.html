021    package xbird.xquery.func.doc;
022    
023    import java.net.URI;
024    import java.net.URISyntaxException;
025    
026    import xbird.util.xml.XMLUtils;
027    import xbird.xquery.DynamicError;
028    import xbird.xquery.XQueryException;
029    import xbird.xquery.dm.value.Item;
030    import xbird.xquery.dm.value.Sequence;
031    import xbird.xquery.dm.value.sequence.ValueSequence;
032    import xbird.xquery.dm.value.xsi.BooleanValue;
033    import xbird.xquery.func.BuiltInFunction;
034    import xbird.xquery.func.FunctionSignature;
035    import xbird.xquery.meta.DynamicContext;
036    import xbird.xquery.misc.QNameTable.QualifiedName;
037    import xbird.xquery.type.Type;
038    import xbird.xquery.type.TypeRegistry;
039    import xbird.xquery.type.xs.BooleanType;
040    
041    /**
042     * fn:doc-available($uri as xs:string?) as xs:boolean.
043     * <DIV lang="en"></DIV>
044     * <DIV lang="ja"></DIV>
045     * 
046     * @author Makoto YUI (yuin405+xbird@gmail.com)
047     * @link http://www.w3.org/TR/xquery-operators/#func-doc-available
048     */
049    public final class DocAvailable extends BuiltInFunction {
050        private static final long serialVersionUID = -5172353587233816241L;
051    
052        public static final String SYMBOL = "fn:doc-available";
053        public static final QualifiedName FUNC_NAME = resolve(SYMBOL);
054    
055        public DocAvailable() {
056            super(FUNC_NAME, BooleanType.BOOLEAN);
057        }
058    
059        protected FunctionSignature[] signatures() {
060            final FunctionSignature[] s = new FunctionSignature[1];
061            s[0] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?") });
062            return s;
063        }
064    
065        public BooleanValue eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
066                throws XQueryException {
067            Item uriItem = argv.getItem(0);
068            if(uriItem.isEmpty()) {
069                return BooleanValue.FALSE;
070            }
071            String uriStr = uriItem.stringValue();
072            // If $uri is not a valid xs:anyURI, an error is raised [err:FODC0005].
073            final URI docuri;
074            try {
075                final String unescaped = XMLUtils.unescapeXML(uriStr);
076                docuri = new URI(unescaped);
077            } catch (URISyntaxException e) {
078                throw new DynamicError("err:FODC0005", "Invalid uri: " + uriStr, e);
079            }
080            if(dynEnv.getDocumentManager().isDocumentAvailable(docuri)) {
081                return BooleanValue.TRUE;
082            } else {
083                return BooleanValue.FALSE;
084            }
085        }
086    
087    }
