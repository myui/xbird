021    package xbird.xquery.func.string;
022    
023    import xbird.util.xml.XMLUtils;
024    import xbird.xquery.XQueryException;
025    import xbird.xquery.dm.value.Item;
026    import xbird.xquery.dm.value.Sequence;
027    import xbird.xquery.dm.value.literal.XString;
028    import xbird.xquery.dm.value.sequence.ValueSequence;
029    import xbird.xquery.func.BuiltInFunction;
030    import xbird.xquery.func.FunctionSignature;
031    import xbird.xquery.meta.DynamicContext;
032    import xbird.xquery.type.Type;
033    import xbird.xquery.type.TypeRegistry;
034    import xbird.xquery.type.xs.StringType;
035    
036    /**
037     * fn:encode-for-uri($uri-part as xs:string?) as xs:string.
038     * <DIV lang="en">
039     * This function should be used to process an xs:string to be used as a path segment in a URI.
040     * It is invertible but not idempotent.
041     * </DIV>
042     * <DIV lang="ja"></DIV>
043     * 
044     * @author Makoto YUI (yuin405+xbird@gmail.com)
045     * @link http://www.w3.org/TR/xquery-operators/#func-encode-for-uri
046     */
047    public final class EncodeForUri extends BuiltInFunction {
048        private static final long serialVersionUID = -8525587551429352739L;
049        
050        public static final String SYMBOL = "fn:encode-for-uri";
051    
052        public EncodeForUri() {
053            super(SYMBOL, StringType.STRING);
054        }
055    
056        protected FunctionSignature[] signatures() {
057            final FunctionSignature[] s = new FunctionSignature[1];
058            s[0] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?") });
059            return s;
060        }
061    
062        public XString eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
063                throws XQueryException {
064            // If $uri-part is the empty sequence, returns the zero-length string.
065            Item firstItem = argv.getItem(0);
066            if(firstItem.isEmpty()) {
067                return XString.valueOf("");
068            }
069            assert (firstItem instanceof XString);
070            String uripart = firstItem.stringValue();
071            String escaped = XMLUtils.escapeUri(uripart, true);
072            return XString.valueOf(escaped);
073        }
074    
075    }
