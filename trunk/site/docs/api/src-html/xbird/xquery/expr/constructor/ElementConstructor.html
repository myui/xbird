060        public ElementConstructor(QualifiedName name) {
061            this(name, new ArrayList<AttributeConstructorBase>(4));

064        public ElementConstructor(QualifiedName name, List<AttributeConstructorBase> atts) {
065            this.name = name;
066            this._attslen = atts.size();
067            this.attributes = atts;
068            this._type = name == null ? ElementTest.ANY_ELEMENT : new ElementTest(name);
069        }

071        public ElementConstructor(XQExpression nameExpr) {
072            this(null, Collections.<AttributeConstructorBase> emptyList());
073            this._nameExpr = nameExpr;
074        }

088        public String getElemName() {
089            StringBuilder buf = new StringBuilder(32);
090            String prefix = name.getPrefix();
091            if(prefix != null && prefix.length() > 0) {
092                buf.append(prefix);
093                buf.append(':');
094            }
095            buf.append(name.getLocalPart());
096            return buf.toString();
097        }

107        public void addContent(XQExpression expr) throws TypeError {
108            if(expr == null) {
109                throw new IllegalArgumentException();
110            }
111            if(expr instanceof AttributeConstructorBase) {
112                if(!contents.isEmpty()) {
113                    throw new TypeError("err:XQTY0024", "an attribute node following a node that is not an attribute node");
114                }
115                AttributeConstructorBase att = (AttributeConstructorBase) expr;
116                if(att instanceof NamespaceConstructor) {
117                    attributes.add(0, att);
118                    ++_attslen;
119                } else if(att instanceof AttributeConstructor) {
120                    attributes.add(att);
121                    ++_attslen;
122                } else {
123                    throw new IllegalStateException("Illegal attribute class: "
124                            + att.getClass().getName());
125                }
126            } else {
127                contents.add(expr);
128            }
129        }

131        @Deprecated
132        public void addContent(TextContent expr) {
133            assert (expr != null);
134            if(expr == TextContent.TrimedText.INSTANCE) {
135                return;
136            }
137            if(contents != null && !contents.isEmpty()) { // tiny optimization.
138                int lastidx = contents.size() - 1;
139                XQExpression lastEntry = contents.get(lastidx);
140                if(lastEntry instanceof TextContent) {
141                    TextContent tc = (TextContent) lastEntry;
142                    if(tc.isCData() && expr.isCData()) {
143                        tc.appendText(expr.getValue().toString());
144                        return;
145                    }
146                }
147            }
148            contents.add(expr);
149        }

154        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
155            if(!_analyzed) {
156                this._analyzed = true;
157                if(name == null) {
158                    if(_nameExpr == null) {
159                        reportError("err:XPTY0004", "name expression was null");
160                    }
161                    this._nameExpr = _nameExpr.staticAnalysis(statEnv);
162                }
163                for(AttributeConstructorBase att : attributes) {
164                    att.staticAnalysis(statEnv);
165                }
166                boolean stripSpace = statEnv.isStripBoundarySpace();
167                int size = contents.size();
168                if(size > 0) {
169                    int from = 0;
170                    {// trim head
171                        XQExpression content = contents.get(0);
172                        if(stripSpace && content instanceof TextContent) {
173                            TextContent tc = (TextContent) content;
174                            TextContent striped = stripBoundaryText(tc);
175                            if(striped == null) {
176                                contents.remove(0);
177                                --size;
178                            } else if(tc != striped) {
179                                contents.set(0, striped);
180                                ++from;
181                            }
182                        }
183                    }
184                    for(int i = from; i < size; i++) {
185                        XQExpression c = contents.get(i);
186                        if(stripSpace && i == (size - 1) && c instanceof TextContent) {// trim end
187                            TextContent tc = (TextContent) c;
188                            TextContent striped = stripBoundaryText(tc);
189                            if(striped == null) {
190                                contents.remove(i);
191                            } else if(tc != striped) {
192                                contents.set(i, striped);
193                            }
194                            break;
195                        }
196                        contents.set(i, c.staticAnalysis(statEnv));
197                    }
198                }
199            }
200            return this;
201        }
202    
203        private static TextContent stripBoundaryText(TextContent tc) {
204            String c = tc.getValue().stringValue();
205            String trimed = c.trim();
206            int trimedLen = trimed.length();
207            if(trimedLen == 0) {
208                return null;
209            }
210            if(trimedLen != c.length()) {
211                return new TextContent(trimed);
212            }
213            return tc;
214        }

216        public Sequence<? extends Item> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
217                throws XQueryException {
218            QualifiedName instName = name;
219            if(instName == null) {
220                instName = processNameExpr(contextSeq, dynEnv);
221            }
222            DMElement dmelem = XQueryDataModel.createElement(instName);
223            boolean hasNamespaceDecl = false;
224            NamespaceBinder knownNamespaces = dynEnv.getStaticContext().getStaticalyKnownNamespaces();
225            // process attribute/namespace
226            for(AttributeConstructorBase att : attributes) {
227                if(att instanceof NamespaceConstructor) {
228                    if(!hasNamespaceDecl) {
229                        knownNamespaces.pushContext(); // TODO REVIEWME performance
230                        hasNamespaceDecl = true;
231                    }
232                    NamespaceConstructor nc = ((NamespaceConstructor) att);
233                    DMNamespace dmns = nc.eval(contextSeq, dynEnv);
234                    dmelem.addNamespace(dmns);
235                    String nsprefix = nc.getPrefix();
236                    String nsuri = nc.getValue();
237                    knownNamespaces.declarePrefix(nsprefix, nsuri);
238                } else if(att instanceof AttributeConstructor) {
239                    AttributeConstructor ac = (AttributeConstructor) att;
240                    DMAttribute dmatt = ac.eval(contextSeq, dynEnv);
241                    dmelem.addAttribute(dmatt);
242                } else {
243                    throw new IllegalStateException("Expected to be NamespaceConstructor or AttributeConstructor instance, but was.. "
244                            + att.getClass().getCanonicalName());
245                }
246            }
247            // process children
248            for(XQExpression content : contents) {
249                Sequence childs = content.eval(contextSeq, dynEnv);
250                if(childs != ValueSequence.EMPTY_SEQUENCE) {
251                    CloneNodesSequence clonedseq = new CloneNodesSequence(childs, dynEnv);            
252                    dmelem.addContents(clonedseq, dynEnv);
253                }
254            }
255            if(hasNamespaceDecl) {
256                knownNamespaces.popContext();
257            }
258            return dmelem;
259        }

261        @Override
262        public void evalAsEvents(XQEventReceiver handler, Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
263                throws XQueryException {
264            QualifiedName instName = name;
265            if(instName == null) {
266                instName = processNameExpr(contextSeq, dynEnv);
267            }
268            handler.evStartElement(instName);
269            boolean hasNamespaceDecl = false;
270            NamespaceBinder knownNamespaces = dynEnv.getStaticContext().getStaticalyKnownNamespaces();
271    
272            final int attlen = _attslen;
273            for(int i = 0; i < attlen; i++) {
274                AttributeConstructorBase att = attributes.get(i);
275                if(att instanceof NamespaceConstructor) {
276                    if(!hasNamespaceDecl) {
277                        knownNamespaces.pushContext(); // TODO REVIEWME performance
278                        hasNamespaceDecl = true;
279                    }
280                    NamespaceConstructor nc = ((NamespaceConstructor) att);
281                    String nsprefix = nc.getPrefix();
282                    String nsuri = nc.getValue();
283                    knownNamespaces.declarePrefix(nsprefix, nsuri);
284                    handler.evNamespace(nsprefix, nsuri);
285                } else {
286                    att.evalAsEvents(handler, contextSeq, dynEnv);
287                }
288            }
289            final boolean old = handler.setBlankRequired(false);
290            final int contentlen = contents.size();
291            for(int i = 0; i < contentlen; i++) {
292                XQExpression content = contents.get(i);
293                content.evalAsEvents(handler, contextSeq, dynEnv);
294                handler.setBlankRequired(true);
295            }
296            handler.setBlankRequired(old);
297    
298            if(hasNamespaceDecl) {
299                knownNamespaces.popContext();
300            }
301            handler.evEndElement(instName);
302        }
