141            this._docid = in.readInt();
142            final IDocumentTable doctbl = (IDocumentTable) in.readObject();
143            if(doctbl instanceof MemoryMappedDocumentTable) {
144                this._mmapedStore = true;
145                MemoryMappedDocumentTable mmDoctbl = (MemoryMappedDocumentTable) doctbl;
146                String docId = mmDoctbl.getDocumentIdentifer();
147                if(docId != null) {
148                    setPool(mmDoctbl, docId);
149                }
150            }
151            this._store = doctbl;

154        private static synchronized void setPool(MemoryMappedDocumentTable mmDoctbl, String docId)
155                throws IOException {
156            ReferenceMap<String, MemoryMappedDocumentTable> cache = _remoteDoctblCache;
157            if(cache == null) {
158                cache = new ReferenceMap<String, MemoryMappedDocumentTable>(ReferenceType.STRONG, ReferenceType.SOFT);
159                _remoteDoctblCache = cache;
160            }
161            MemoryMappedDocumentTable prevDoctbl = cache.putIfAbsent(docId, mmDoctbl);
162            if(prevDoctbl != null) {
163                mmDoctbl.setBufferPool(prevDoctbl.getBufferPool());
164            } else {
165                File tmpDir = (TMP_DATA_DIR == null) ? null : new File(TMP_DATA_DIR);
166                String docname = FileUtils.basename(docId);
167                File tmpFile = File.createTempFile(docname, ".tmp", tmpDir);
168                tmpFile.deleteOnExit();
169                ILongCache<int[]> pool = new DiskPagedLongCache<int[]>(tmpFile, MemoryMappedDocumentTable.CACHED_PAGES, new IntCodec());
170                mmDoctbl.setBufferPool(pool);
171            }
172        }

202        public <T extends DTMNodeBase> T createNode(byte kind, long id) {
203            switch(kind) {
204                case NodeKind.DOCUMENT:
205                    return (T) new DTMDocument(this, id);
206                case NodeKind.ELEMENT:
207                    return (T) new DTMElement(this, id);
208                case NodeKind.ATTRIBUTE:
209                    return (T) new DTMAttribute(this, id);
210                case NodeKind.NAMESPACE:
211                    return (T) new DTMNamespace(this, id);
212                case NodeKind.COMMENT:
213                    return (T) new DTMComment(this, id);
214                case NodeKind.TEXT:
215                    return (T) new DTMText(this, id);
216                case NodeKind.PROCESSING_INSTRUCTION:
217                    return (T) new DTMProcessingInstruction(this, id);
218                default:
219                    throw new IllegalStateException("Invalid node kind: " + kind + " for node#" + id);
220            }
221        }

256        public void export(final long nodeid, final XQEventReceiver receiver) throws XQueryException {
257            final PagingProfile profile = _store.getPagingProfile();
258            Strategy origStrategy = null;
259            if(profile != null) {
260                origStrategy = profile.getStrategy();
261                profile.setStrategy(Strategy.serialization);
262            }
263            final IDocumentTable doctbl = _store;
264            switch(doctbl.getNodeKindAt(nodeid)) {
265                case NodeKind.DOCUMENT:
266                    receiver.evStartDocument();
267                    final long firstChild = doctbl.firstChild(nodeid);
268                    if(firstChild != -1L) {
269                        export(firstChild, receiver);
270                        long nextSib = doctbl.nextSibling(firstChild);
271                        while(nextSib != 0L) {
272                            export(nextSib, receiver);
273                            nextSib = doctbl.nextSibling(firstChild);
274                        }
275                    }
276                    receiver.evEndDocument();
277                    break;
278                case NodeKind.ELEMENT:
279                    final QualifiedName qname = doctbl.getName(nodeid);
280                    receiver.evStartElement(nodeid, qname);
281                    // namespace decl
282                    final int nsdeclCnt = doctbl.getNamespaceCountAt(nodeid);
283                    for(int i = 0; i < nsdeclCnt; i++) {
284                        final long nsid = doctbl.getNamespaceDecl(nodeid, i);
285                        final QualifiedName nsName = doctbl.getAttributeName(nsid);
286                        final String uri = doctbl.getText(nsid);
287                        receiver.evNamespace(nsid, nsName, uri);
288                    }
289                    // attribute
290                    final int attrCnt = doctbl.getAttributeCountAt(nodeid);
291                    for(int i = 0; i < attrCnt; i++) {
292                        final long attid = doctbl.getAttribute(nodeid, i);
293                        final QualifiedName attName = doctbl.getAttributeName(attid);
294                        final String attValue = doctbl.getText(attid);
295                        receiver.evAttribute(attid, attName, attValue);
296                    }
297                    final long elemFirstChild = doctbl.firstChild(nodeid);
298                    if(elemFirstChild != -1L) {
299                        export(elemFirstChild, receiver);
300                        long nextSib = doctbl.nextSibling(elemFirstChild);
301                        while(nextSib != 0L) {
302                            export(nextSib, receiver);
303                            nextSib = doctbl.nextSibling(nextSib);
304                        }
305                    }
306                    receiver.evEndElement(nodeid, qname);
307                    break;
308                case NodeKind.ATTRIBUTE:
309                    receiver.evAttribute(nodeid, doctbl.getAttributeName(nodeid), doctbl.getText(nodeid));
310                    break;
311                case NodeKind.NAMESPACE:
312                    receiver.evNamespace(nodeid, doctbl.getAttributeName(nodeid), doctbl.getText(nodeid));
313                    break;
314                case NodeKind.TEXT:
315                    receiver.evText(nodeid, doctbl.getText(nodeid));
316                    break;
317                case NodeKind.COMMENT:
318                    receiver.evComment(nodeid, doctbl.getText(nodeid));
319                    break;
320                case NodeKind.PROCESSING_INSTRUCTION:
321                    final QualifiedName pi = doctbl.getName(nodeid);
322                    receiver.evProcessingInstruction(nodeid, pi.getLocalPart(), pi.getNamespaceURI());
323                    break;
324                default:
325                    throw new IllegalStateException("Invalid node kind '"
326                            + NodeKind.resolveName(doctbl.getNodeKindAt(nodeid)) + "' for node#"
327                            + nodeid);
328            }
329            if(profile != null) {
330                profile.setStrategy(origStrategy);
331            }
332        }

334        public void loadDocument(InputStream is) throws XQueryException {
335            if(!(is instanceof FastBufferedInputStream)) {
336                is = new FastBufferedInputStream(is);
337            }
338            _handler.init();
339            final StopWatch sw = new StopWatch();
340            try {
341                _reader.parse(new InputSource(is));
342            } catch (IOException ie) {
343                throw new DynamicError("Invalid source", ie);
344            } catch (SAXException se) {
345                throw new DynamicError("Parse failed", se);
346            }
347            if(LOG.isDebugEnabled()) {
348                LOG.debug("Elasped time of building a DTM: " + sw.elapsed() + " msec");
349            }
350            this._docid = nextDocumentId();
351            if(_store instanceof IProfiledDocumentTable) {
352                ((IProfiledDocumentTable) _store).setProfiling(true);
353            }
354        }

361        private static final XMLReader getXMLReader(final DocumentTableBuilder handler, final boolean parseAsHtml) {
362            final XMLReader myReader;
363            try {
364                if(parseAsHtml) {
365                    Class clazz = ClassResolver.get(HTML_PARSER_CLASS);
366                    assert (clazz != null);
367                    myReader = ObjectUtils.<XMLReader> instantiate(clazz);
368                } else {
369                    SAXParserFactory factory = SAXParserFactory.newInstance();
370                    factory.setNamespaceAware(true);
371                    SAXParser parser = factory.newSAXParser();
372                    myReader = parser.getXMLReader();
373                }
374            } catch (Exception e) {
375                throw new XQRTException("Creating SAX XMLReader failed", e);
376            }
377            // setup handlers (requires saxHandler)
378            myReader.setContentHandler(handler);
379            try {
380                myReader.setProperty("http://xml.org/sax/properties/lexical-handler", handler);
381                myReader.setFeature("http://xml.org/sax/features/validation", true);
382                if(!parseAsHtml) {
383                    myReader.setFeature("http://apache.org/xml/features/validation/dynamic", true);
384                    myReader.setFeature("http://apache.org/xml/features/validation/schema", true);
385                }
386            } catch (Exception e) {
387                throw new XQRTException("Configuaring SAX XMLReader failed.", e);
388            }
389            // setup entity resolver
390            if(resolveEntity) {
391                org.apache.xml.resolver.CatalogManager catalog = org.apache.xml.resolver.CatalogManager.getStaticManager();
392                catalog.setIgnoreMissingProperties(true);
393                EntityResolver resolver = new org.apache.xml.resolver.tools.CatalogResolver(catalog);
394                myReader.setEntityResolver(resolver);
395            }
396            return myReader;
397        }

435                assert (_toReplace == null);
436                final XQEventDecoder decoder = new XQEventDecoder(in);
437                final Sequence<Item> decoded;
438                try {

440                } catch (IOException ioe) {

442                } catch (XQueryException xqe) {

446                }
447                this._toReplace = decoded;

450            public void writeTo(final ObjectOutput out, final SerializationContext context)
451                    throws IOException {
452                out.writeObject(_model);
453                out.writeByte(nodeKind());
454                long curr = getPosition();
455                out.writeLong(curr);
456                long nextsib = _store.nextSibling(_id);
457                long last = (nextsib == 0L) ? curr : nextsib;
458                out.writeLong(last);
459                long[] textBlocks = ((MemoryMappedDocumentTable) _store).referredTextBlocks(curr, last, context);
460                out.writeInt(textBlocks.length);
461                for(long tb : textBlocks) {
462                    out.writeLong(tb);
463                }
464            }

466            public static DTMNodeBase readFrom(final ObjectInput in, SerializationContext serContext)
467                    throws IOException, ClassNotFoundException {
468                DocumentTableModel model = (DocumentTableModel) in.readObject();
469                byte nodeKind = in.readByte();
470                long curr = in.readLong();
471                DTMNodeBase node = getPrototype(model, nodeKind, curr);
472                long last = in.readLong();
473                final int tbSize = in.readInt();
474                final long[] textBlocks = new long[tbSize];
475                for(int i = 0; i < tbSize; i++) {
476                    textBlocks[i] = in.readLong();
477                }
478                MemoryMappedDocumentTable mmtable = (MemoryMappedDocumentTable) node._store;
479                mmtable.markReferredBlocks(curr, last, textBlocks, serContext);
480                return node;
481            }

483            public static void redirect(ObjectInput in, ObjectOutput out) throws IOException,
484                    ClassNotFoundException {
485                DocumentTableModel model = (DocumentTableModel) in.readObject();
486                out.writeObject(model);
487                byte nodeKind = in.readByte();
488                out.writeByte(nodeKind);
489                long curr = in.readLong();
490                out.writeLong(curr);
491                long last = in.readLong();
492                out.writeLong(last);
493                final int tbsize = in.readInt();
494                out.writeInt(tbsize);
495                final byte[] b = new byte[80];
496                int i = 0;
497                final int limit = tbsize - 9;
498                for(; i < limit; i += 10) {
499                    in.readFully(b, 0, 80);
500                    out.write(b, 0, 80);
501                }
502                for(; i < tbsize; i++) {
503                    in.readFully(b, 0, 8);
504                    out.write(b, 0, 8);
505                }
506            }

557            @Override
558            public final XQNode following(final boolean firstcall) {
559                final PagingProfile profile = _store.getPagingProfile();
560                final XQNode node;
561                if(profile != null) {
562                    Strategy origStrategy = profile.getStrategy();
563                    if(origStrategy != Strategy.nextsib) {
564                        assert (origStrategy != null);
565                        profile.setStrategy(Strategy.nextsib);
566                        node = super.following(firstcall);
567                        profile.setStrategy(origStrategy);
568                    } else {
569                        node = super.following(firstcall);
570                    }
571                } else {
572                    node = super.following(firstcall);
573                }
574                return node;
575            }

583                final IDocumentTable store = documentTable();
584                final long nextsib = store.nextSibling(_id);
585                if(nextsib == 0L) {

592            public final DTMNodeBase nextSibling(NodeTest filter) {
593                if(_id == 0L) {
594                    return null;
595                }
596                final IDocumentTable store = documentTable();
597                final DocumentTableModel model = getDataModel();
598                for(long nid = store.nextSibling(_id); nid != 0L; nid = store.nextSibling(nid)) {
599                    byte nodekind = store.getNodeKindAt(nid);
600                    DTMNodeBase probe = getPrototype(model, nodekind, nid);
601                    if(filter.accepts(probe)) {
602                        return model.createNode(probe.nodeKind(), probe.getPosition());
603                    }
604                }
605                return null;
606            }

682                    while(p != null) {
683                        String uri = p.baseUri();

687                        p = p.parent();

710            public DTMAttribute getAttribute(String nsuri, String attname) {
711                assert (nsuri != null && attname != null);
712                final IDocumentTable store = documentTable();
713                int attcnt = store.getAttributeCountAt(_id);
714                for(int i = 0; i < attcnt; i++) {
715                    long attid = store.getAttribute(_id, i);
716                    assert (attid != -1);
717                    assert (store.getNodeKindAt(attid) != NodeKind.ATTRIBUTE);
718                    QualifiedName qname = store.getAttributeName(attid);
719                    String attUri = qname.getNamespaceURI();
720                    String attName = qname.getLocalPart();
721                    if(nsuri != null && nsuri.equals(attUri)) {
722                        if(attname.equals(attName)) {
723                            final DocumentTableModel model = getDataModel();
724                            return new DTMAttribute(model, attid);
725                        }
726                    }
727                }
728                return null;
729            }

772            public boolean nilled() {
773                final IDocumentTable store = documentTable();
774                long attid;
775                for(int i = 0; (attid = getAttribute(i)) != -1; i++) {
776                    assert (attid != -1);
777                    byte attkind = store.getNodeKindAt(attid);
778                    if(attkind == NodeKind.ATTRIBUTE) {
779                        QualifiedName qname = store.getAttributeName(attid);
780                        String nsuri = qname.getNamespaceURI();
781                        String attname = qname.getLocalPart();
782                        if(XQueryConstants.XSI_URI.equals(nsuri) && "nil".equals(attname)) {
783                            String attval = store.getText(attid);
784                            if(attval == null) {
785                                return false;
786                            }
787                            return BooleanValue.toBoolean(attval);
788                        }
789                    }
790                }
791                return false;
792            }
