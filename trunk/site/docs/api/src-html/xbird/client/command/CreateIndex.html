040        private static final String[] COMMAND = new String[] { "create", null, "index" };
041        private static final String[] IDXTYPE = new String[] { "value", "path", "fulltext" };
042        private static final int VALUE_IDX = 0, PATH_IDX = 1, FULLTEXT_IDX = 2;
043        private static final String PATTERN_CMD = "on", NAMESPACE_CMD = "nsmap", DATATYPE_CMD = "as";
044    
045        public CreateIndex(Session session) {
046            super(session);
047        }
048    
049        public boolean match(String[] args) {
050            if(args.length < COMMAND.length) {
051                return false;
052            }
053            if(!(COMMAND[0].equals(args[0]) & COMMAND[2].equals(args[2]))) {
054                return false;
055            }
056            if(!ArrayUtils.contains(IDXTYPE, args[1])) {
057                return false;
058            }
059            return true;
060        }
061    
062        public boolean process(String[] args) throws CommandFailedException {
063            final DbCollection col = session.getContextCollection();
064            if(col == null) {
065                throwException("Context collection is not set");
066            }
067            final int type = ArrayUtils.indexOf(IDXTYPE, args[1]);
068            assert (type != ArrayUtils.INDEX_NOT_FOUND);
069            final boolean status;
070            switch(type) {
071                case PATH_IDX: {
072                    final String[] result = parsePattern(args, PATTERN_CMD, NAMESPACE_CMD);
073                    status = createPathIndex(col, result[0], result[1]);
074                    break;
075                }
076                case VALUE_IDX: {
077                    final String[] result = parsePattern(args, DATATYPE_CMD, PATTERN_CMD, NAMESPACE_CMD);
078                    status = createValueIndex(col, result[0], result[1], result[2]);
079                    break;
080                }
081                case FULLTEXT_IDX: {
082                    final String[] result = parsePattern(args, PATTERN_CMD, NAMESPACE_CMD);
083                    status = createFullTextIndex(col, result[0], result[1]);
084                    break;
085                }
086                default:
087                    throw new IllegalStateException("Illegal type: " + type);
088            }
089            return status;
090        }

107        private static NamespaceContext toNamespaceContexts(final String nsmap) {
108            final NamespaceBinder namespaces = new NamespaceBinder();
109            if(nsmap != null) {
110                final String[] entries = nsmap.split(",");
111                for(String entry : entries) {
112                    String[] items = entry.split("=");
113                    if(items.length != 2) {
114                        throw new IllegalArgumentException("Illegal nsmap description: " + nsmap);
115                    }
116                    namespaces.declarePrefix(items[0], items[1]);
117                }
118            }
119            return namespaces;
120        }
121    
122        private static String[] parsePattern(final String[] args, final String... pattern) {
123            final int ptnlen = pattern.length;
124            final String[] result = new String[ptnlen];
125            int ri = 0;
126            final int arglen = args.length;
127            for(int i = COMMAND.length; i + 1 < arglen; i += 2) {
128                final String s = args[i];
129                final int pi = ArrayUtils.indexOf(pattern, s, ri);
130                if(pi == ArrayUtils.INDEX_NOT_FOUND) {
131                    continue;
132                }
133                result[pi] = args[i + 1];
134                ri = pi + 1;
135                if(ri >= ptnlen) {
136                    break;
137                }
138            }
139            return result;
140        }
