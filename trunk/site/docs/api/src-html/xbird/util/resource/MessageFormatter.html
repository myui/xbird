021    package xbird.util.resource;
022    
023    import java.io.IOException;
024    import java.io.InputStream;
025    import java.text.MessageFormat;
026    import java.util.*;
027    
028    import xbird.XBirdError;
029    
030    /**
031     * 
032     * <DIV lang="en"></DIV>
033     * <DIV lang="ja"></DIV>
034     * 
035     * @author Makoto YUI (yuin405+xbird@gmail.com)
036     */
037    public final class MessageFormatter {
038    
039        private final Map<String, String> bundle;
040    
041        public MessageFormatter(InputStream is) {
042            final ResourceBundle b;
043            try {
044                b = new PropertyResourceBundle(is);
045            } catch (IOException e) {
046                throw new XBirdError(e);
047            }
048            bundle = convertMap(b);
049        }
050    
051        public String getMessage(String msgCode) {
052            return getMessage(msgCode, null);
053        }
054    
055        public String getMessage(String msgCode, Object[] args) {
056            String pattern = bundle.get(msgCode);
057            return (pattern == null) ? msgCode : MessageFormat.format(pattern, args);
058        }
059    
060        private static final Map<String, String> convertMap(ResourceBundle bundle) {
061            Map<String, String> ret = new HashMap<String, String>();
062            for(Enumeration e = bundle.getKeys(); e.hasMoreElements();) {
063                String key = (String) e.nextElement();
064                String value = bundle.getString(key);
065                ret.put(key, value);
066            }
067            return ret;
068        }
069    
070    }
