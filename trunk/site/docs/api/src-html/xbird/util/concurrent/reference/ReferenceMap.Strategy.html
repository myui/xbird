056    /**
057     * Concurrent hash map that wraps keys and/or values in soft or weak references.
058     * Does not support null keys or values. Uses identity equality for weak and
059     * soft keys.
060     *
061     * <p>The concurrent semantics of {@link ConcurrentHashMap} combined with the
062     * fact that the garbage collector can asynchronously reclaim and clean up
063     * keys and values at any time can lead to some racy semantics. For example,
064     * {@link #size()} returns an upper bound on the size; that is, the actual size
065     * may be smaller in cases where the key or value has been reclaimed but the map
066     * entry has not been cleaned up yet.
067     *
068     * @author crazybob@google.com (Bob Lee)
069     * @author fry@google.com (Charles Fry)
070     * @author Makoto YUI (yuin405+xbird@gmail.com)
071     */
072    @SuppressWarnings("unchecked")
073    public class ReferenceMap<K, V> extends AbstractMap<K, V>
074            implements ConcurrentMap<K, V>, Serializable {
075    
076        transient ConcurrentMap<Object, Object> delegate;
077    
078        private final ReferenceType keyReferenceType;
079        private final ReferenceType valueReferenceType;
080    
081        /**
082         * Concurrent hash map that wraps keys and/or values based on specified
083         * reference types.
084         *
085         * @param delegateMap internally used map instance
086         * @param keyReferenceType key reference type
087         * @param valueReferenceType value reference type
088         */
089        public ReferenceMap(ConcurrentMap delegateMap, ReferenceType keyReferenceType, ReferenceType valueReferenceType) {
090            ensureNotNull(keyReferenceType, valueReferenceType);
091    
092            if(keyReferenceType == ReferenceType.PHANTOM || valueReferenceType == ReferenceType.PHANTOM) {
093                throw new IllegalArgumentException("Phantom references not supported.");
094            }
095    
096            this.delegate = delegateMap;
097            this.keyReferenceType = keyReferenceType;
098            this.valueReferenceType = valueReferenceType;
099        }
100    
101        /**
102         * Concurrent hash map that wraps keys and/or values based on specified
103         * reference types.
104         *
105         * @param keyReferenceType key reference type
106         * @param valueReferenceType value reference type
107         */
108        public ReferenceMap(ReferenceType keyReferenceType, ReferenceType valueReferenceType) {
109            this(new ConcurrentHashMap(), keyReferenceType, valueReferenceType);
110        }
111    
112        V internalGet(K key) {
113            Object valueReference = delegate.get(makeKeyReferenceAware(key));
114            return dereferenceValue(valueReference);
115        }
116    
117        public V get(final Object key) {
118            ensureNotNull(key);
119            return internalGet((K) key);
120        }
121    
122        private V execute(Strategy strategy, K key, V value) {
123            ensureNotNull(key, value);
124            Object keyReference = referenceKey(key);
125            return (V) strategy.execute(this, keyReference, referenceValue(keyReference, value));
126        }
127    
128        public V put(K key, V value) {
129            return execute(putStrategy(), key, value);
130        }
131    
132        public V remove(Object key) {
133            ensureNotNull(key);
134            Object referenceAwareKey = makeKeyReferenceAware(key);
135            Object valueReference = delegate.remove(referenceAwareKey);
136            return dereferenceValue(valueReference);
137        }
138    
139        public int size() {
140            return delegate.size();
141        }
142    
143        public boolean isEmpty() {
144            return delegate.isEmpty();
145        }
146    
147        public boolean containsKey(Object key) {
148            ensureNotNull(key);
149            Object referenceAwareKey = makeKeyReferenceAware(key);
150            return delegate.containsKey(referenceAwareKey);
151        }
152    
153        public boolean containsValue(Object value) {
154            ensureNotNull(value);
155            for(Object valueReference : delegate.values()) {
156                if(value.equals(dereferenceValue(valueReference))) {
157                    return true;
158                }
159            }
160            return false;
161        }
162    
163        public void putAll(Map<? extends K, ? extends V> t) {
164            for(Map.Entry<? extends K, ? extends V> entry : t.entrySet()) {
165                put(entry.getKey(), entry.getValue());
166            }
167        }
168    
169        public void clear() {
170            delegate.clear();
171        }
172    
173        public V putIfAbsent(K key, V value) {
174            return execute(putIfAbsentStrategy(), key, value);
175        }
176    
177        public boolean remove(Object key, Object value) {
178            ensureNotNull(key, value);
179            Object referenceAwareKey = makeKeyReferenceAware(key);
180            Object referenceAwareValue = makeValueReferenceAware(value);
181            return delegate.remove(referenceAwareKey, referenceAwareValue);
182        }
183    
184        public boolean replace(K key, V oldValue, V newValue) {
185            ensureNotNull(key, oldValue, newValue);
186            Object keyReference = referenceKey(key);
187    
188            Object referenceAwareOldValue = makeValueReferenceAware(oldValue);
189            return delegate.replace(keyReference, referenceAwareOldValue, referenceValue(keyReference, newValue));
190        }
191    
192        public V replace(K key, V value) {
193            return execute(replaceStrategy(), key, value);
194        }

196        private transient volatile Set<Map.Entry<K, V>> entrySet = null;
197    
198        public Set<Map.Entry<K, V>> entrySet() {
199            if(entrySet == null) {
200                entrySet = new EntrySet();
201            }
202            return entrySet;
203        }

205        /** Dereferences an entry. Returns null if the key or value has been gc'ed. */
206        private Entry dereferenceEntry(Map.Entry<Object, Object> entry) {
207            K key = dereferenceKey(entry.getKey());
208            V value = dereferenceValue(entry.getValue());
209            return (key == null || value == null) ? null : new Entry(key, value);
210        }
211    
212        /** Creates a reference for a key. */
213        Object referenceKey(K key) {
214            switch(keyReferenceType) {
215                case STRONG:
216                    return key;
217                case SOFT:
218                    return new SoftKeyReference(key);
219                case WEAK:
220                    return new WeakKeyReference(key);
221                default:
222                    throw new AssertionError();
223            }
224        }
225    
226        /** Converts a reference to a key. */
227        K dereferenceKey(Object o) {
228            return (K) dereference(keyReferenceType, o);
229        }
230    
231        /** Converts a reference to a value. */
232        V dereferenceValue(Object o) {
233            if(o == null) {
234                return null;
235            }
236            Object value = dereference(valueReferenceType, o);
237            if(o instanceof InternalReference) {
238                InternalReference ref = (InternalReference) o;
239                if(value == null) {
240                    // old value was garbage collected
241                    ref.finalizeReferent();
242                }
243            }
244            return (V) value;
245        }
246    
247        /** Returns the refererent for reference given its reference type. */
248        private Object dereference(ReferenceType referenceType, Object reference) {
249            return referenceType == STRONG ? reference : ((Reference) reference).get();
250        }
251    
252        /** Creates a reference for a value. */
253        Object referenceValue(Object keyReference, V value) {
254            switch(valueReferenceType) {
255                case STRONG:
256                    return value;
257                case SOFT:
258                    return new SoftValueReference(keyReference, value);
259                case WEAK:
260                    return new WeakValueReference(keyReference, value);
261                default:
262                    throw new AssertionError();
263            }
264        }
265    
266        /**
267         * Wraps key so it can be compared to a referenced key for equality.
268         */
269        private Object makeKeyReferenceAware(Object o) {
270            return keyReferenceType == STRONG ? o : new KeyReferenceAwareWrapper(o);
271        }
272    
273        /** Wraps value so it can be compared to a referenced value for equality. */
274        private Object makeValueReferenceAware(Object o) {
275            return valueReferenceType == STRONG ? o : new ReferenceAwareWrapper(o);
276        }

278        /**
279         * Marker interface to differentiate external and internal references. Also
280         * duplicates FinalizableReference and Reference.get for internal use.
281         */
282        interface InternalReference {
283            void finalizeReferent();
284    
285            Object get();
286        }
287    
288        private static int keyHashCode(Object key) {
289            return System.identityHashCode(key);
290        }
291    
292        /*
293         * Tests weak and soft references for identity equality. Compares references
294         * to other references and wrappers. If o is a reference, this returns true if
295         * r == o or if r and o reference the same non-null object. If o is a wrapper,
296         * this returns true if r's referent is identical to the wrapped object.
297         */
298        protected static boolean referenceEquals(Reference r, Object o) {
299            // compare reference to reference.
300            if(o instanceof InternalReference) {
301                // are they the same reference? used in cleanup.
302                if(o == r) {
303                    return true;
304                }
305    
306                // do they reference identical values? used in conditional puts.
307                Object referent = ((Reference) o).get();
308                return referent != null && referent == r.get();
309            }
310    
311            // is the wrapped object identical to the referent? used in lookups.
312            return ((ReferenceAwareWrapper) o).unwrap() == r.get();
313        }
314    
315        /**
316         * Returns {@code true} if the specified value reference has been garbage
317         * collected. The value behind the reference is also passed in, rather than
318         * queried inside this method, to ensure that the return statement of this
319         * method will still hold true after it has returned (that is, a value
320         * reference exists outside of this method which will prevent that value from
321         * being garbage collected).
322         *
323         * @param valueReference the value reference to be tested
324         * @param value the object referenced by {@code valueReference}
325         * @return {@code true} if {@code valueReference} is non-null and {@code
326         *     value} is null
327         */
328        private static boolean isExpired(Object valueReference, Object value) {
329            return (valueReference != null) && (value == null);
330        }

332        /**
333         * Big hack. Used to compare keys and values to referenced keys and values
334         * without creating more references.
335         */
336        static class ReferenceAwareWrapper {
337            final Object wrapped;
338    
339            ReferenceAwareWrapper(Object wrapped) {
340                this.wrapped = wrapped;
341            }
342    
343            Object unwrap() {
344                return wrapped;
345            }
346    
347            public int hashCode() {
348                return wrapped.hashCode();
349            }
350    
351            public boolean equals(Object obj) {
352                // defer to reference's equals() logic.
353                return obj.equals(this);
354            }
355        }

357        /** Used for keys. Overrides hash code to use identity hash code. */
358        static class KeyReferenceAwareWrapper extends ReferenceAwareWrapper {
359            public KeyReferenceAwareWrapper(Object wrapped) {
360                super(wrapped);
361            }
362    
363            public int hashCode() {
364                return System.identityHashCode(wrapped);
365            }
366        }

467        private enum PutStrategy implements Strategy {
468            PUT {
469                public Object execute(ReferenceMap map, Object keyReference, Object valueReference) {
470                    return map.dereferenceValue(map.delegate.put(keyReference, valueReference));
471                }
472            },
473    
474            REPLACE {
475                public Object execute(ReferenceMap map, Object keyReference, Object valueReference) {
476                    // ensure that the existing value is not collected
477                    do {
478                        Object existingValueReference;
479                        Object existingValue;
480                        do {
481                            existingValueReference = map.delegate.get(keyReference);
482                            existingValue = map.dereferenceValue(existingValueReference);
483                        } while(isExpired(existingValueReference, existingValue));
484    
485                        if(existingValueReference == null) {
486                            // nothing to replace
487                            return false;
488                        }
489    
490                        if(map.delegate.replace(keyReference, existingValueReference, valueReference)) {
491                            // existingValue didn't expire since we still have a reference to it
492                            return existingValue;
493                        }
494                    } while(true);
495                }
496            },
497    
498            PUT_IF_ABSENT {
499                public Object execute(ReferenceMap map, Object keyReference, Object valueReference) {
500                    Object existingValueReference;
501                    Object existingValue;
502                    do {
503                        existingValueReference = map.delegate.putIfAbsent(keyReference, valueReference);
504                        existingValue = map.dereferenceValue(existingValueReference);
505                    } while(isExpired(existingValueReference, existingValue));
506                    return existingValue;
507                }
508            },
509        }
510    
511        private static PutStrategy defaultPutStrategy;
512    
513        protected PutStrategy getPutStrategy() {
514            return defaultPutStrategy;
515        }

517        class Entry implements Map.Entry<K, V> {
518            final K key;
519            V value;
520    
521            public Entry(K key, V value) {
522                this.key = key;
523                this.value = value;
524            }
525    
526            public K getKey() {
527                return this.key;
528            }
529    
530            public V getValue() {
531                return this.value;
532            }
533    
534            public V setValue(V newValue) {
535                value = newValue;
536                return put(key, newValue);
537            }
538    
539            public int hashCode() {
540                return key.hashCode() * 31 + value.hashCode();
541            }
542    
543            public boolean equals(Object o) {
544                if(!(o instanceof ReferenceMap.Entry)) {
545                    return false;
546                }
547    
548                Entry entry = (Entry) o;
549                return key.equals(entry.key) && value.equals(entry.value);
550            }
551    
552            public String toString() {
553                return key + "=" + value;
554            }
555        }

557        private class ReferenceIterator implements Iterator<Map.Entry<K, V>> {
558            private Iterator<Map.Entry<Object, Object>> i = delegate.entrySet().iterator();
559            private Map.Entry<K, V> nextEntry;
560            private Map.Entry<K, V> lastReturned;
561    
562            public ReferenceIterator() {
563                advance();
564            }
565    
566            private void advance() {
567                while(i.hasNext()) {
568                    Map.Entry<K, V> entry = dereferenceEntry(i.next());
569                    if(entry != null) {
570                        nextEntry = entry;
571                        return;
572                    }
573                }
574    
575                // nothing left
576                nextEntry = null;
577            }
578    
579            public boolean hasNext() {
580                return nextEntry != null;
581            }
582    
583            public Map.Entry<K, V> next() {
584                if(nextEntry == null) {
585                    throw new NoSuchElementException();
586                }
587                lastReturned = nextEntry;
588                advance();
589                return lastReturned;
590            }
591    
592            public void remove() {
593                ReferenceMap.this.remove(lastReturned.getKey());
594            }
595        }

597        private class EntrySet extends AbstractSet<Map.Entry<K, V>> {
598            public Iterator<Map.Entry<K, V>> iterator() {
599                return new ReferenceIterator();
600            }
601    
602            public int size() {
603                return delegate.size();
604            }
605    
606            public boolean contains(Object o) {
607                if(!(o instanceof Map.Entry)) {
608                    return false;
609                }
610                Map.Entry<K, V> e = (Map.Entry<K, V>) o;
611                V v = ReferenceMap.this.get(e.getKey());
612                return v != null && v.equals(e.getValue());
613            }
614    
615            public boolean remove(Object o) {
616                if(!(o instanceof Map.Entry)) {
617                    return false;
618                }
619                Map.Entry<K, V> e = (Map.Entry<K, V>) o;
620                return ReferenceMap.this.remove(e.getKey(), e.getValue());
621            }
622    
623            public void clear() {
624                delegate.clear();
625            }
626        };

636            for(int i = 0; i < array.length; i++) {

643        private void writeObject(ObjectOutputStream out) throws IOException {
644            out.defaultWriteObject();
645            out.writeInt(size());
646            for(Map.Entry<Object, Object> entry : delegate.entrySet()) {
647                Object key = dereferenceKey(entry.getKey());
648                Object value = dereferenceValue(entry.getValue());
649    
650                // don't persist gc'ed entries.
651                if(key != null && value != null) {
652                    out.writeObject(key);
653                    out.writeObject(value);
654                }
655            }
656            out.writeObject(null);
657        }
658    
659        private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
660            in.defaultReadObject();
661            int size = in.readInt();
662            this.delegate = new ConcurrentHashMap<Object, Object>(size);
663            while(true) {
664                K key = (K) in.readObject();
665                if(key == null) {
666                    break;
667                }
668                V value = (V) in.readObject();
669                put(key, value);
670            }
671        }
