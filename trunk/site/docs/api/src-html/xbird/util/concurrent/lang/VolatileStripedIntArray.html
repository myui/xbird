013    package xbird.util.concurrent.lang;
014    
015    import java.io.Serializable;
016    import java.util.Arrays;
017    
018    import xbird.util.lang.HashUtils;
019    import xbird.util.lang.SystemUtils;
020    import xbird.util.lang.UnsafeUtils;
021    
022    /**
023     * 
024     * <DIV lang="en"></DIV>
025     * <DIV lang="ja"></DIV>
026     * 
027     * @author Makoto YUI (yuin405+xbird@gmail.com)
028     * @see http://www.cs.umd.edu/~pugh/java/memoryModel/archive/index.html#332
029     */
030    public final class VolatileStripedIntArray implements Serializable {
031        private static final long serialVersionUID = -1666971843031968502L;
032    
033        private static final sun.misc.Unsafe unsafe = UnsafeUtils.getUnsafe();
034        private static final int base = unsafe.arrayBaseOffset(int[].class);
035        private static final int scale = unsafe.arrayIndexScale(int[].class);
036    
037        private final int[] array;
038        private final int numDims;
039        private final int mask;
040        /** capacity of each dimension */
041        private final int capacity;
042    
043        /** used for barrier */
044        private volatile int macguffin;
045    
046        public VolatileStripedIntArray(int cap) {
047            this(cap, SystemUtils.is64BitVM() ? 8 : 4);
048        }
049    
050        public VolatileStripedIntArray(int cap, int dim) {
051            assert (HashUtils.isPowerOfTwo(cap)) : cap;
052            assert (HashUtils.isPowerOfTwo(dim)) : dim;
053            if(cap < dim) {
054                throw new IllegalArgumentException("cap must be greater than or equals to " + dim
055                        + ", but was .. " + cap);
056            }
057            this.array = new int[cap];
058            this.numDims = dim;
059            this.mask = dim - 1;
060            this.capacity = cap / dim;
061            this.macguffin = 0;
062        }
063    
064        public int get(final int i) {
065            int idx = toPhysicalIndex(i);
066            return array[idx + macguffin];
067        }
068    
069        public void set(final int i, final int value) {
070            int idx = toPhysicalIndex(i);
071            array[idx] = value;
072            macguffin = 0;
073        }
074    
075        public boolean compareAndSet(final int idx, final int expect, final int update) {
076            int phyidx = toPhysicalIndex(idx);
077            return unsafe.compareAndSwapObject(array, rawIndex(phyidx), expect, update);
078        }
079    
080        private int toPhysicalIndex(final int logical) {
081            int remainder = logical & mask;
082            int quotient = logical / numDims;
083            return remainder * capacity + quotient;
084        }
085    
086        private long rawIndex(final int i) {
087            assert (i >= 0 && i < array.length) : "index: " + i;
088            return base + i * scale;
089        }
090    
091        public static void main(String[] args) {
092            final int estsize = 32;
093            VolatileStripedIntArray ary = new VolatileStripedIntArray(estsize, 8);
094            for(int i = 0; i < estsize; i++) {
095                ary.set(i, i);
096            }
097            for(int i = 0; i < estsize; i++) {
098                if(ary.get(i) != i) {
099                    System.err.println("error found at " + i);
100                    System.exit(1);
101                }
102            }
103            System.err.println("succeed!");
104            System.out.println(Arrays.toString(ary.array));
105        }
106    }
