013    package xbird.util.concurrent.lang;
014    
015    import java.util.concurrent.atomic.AtomicLong;
016    
017    /**
018     * 
019     * <DIV lang="en"></DIV>
020     * <DIV lang="ja"></DIV>
021     * 
022     * @author Makoto YUI (yuin405+xbird@gmail.com)
023     */
024    public final class AtomicSimpleRandom {
025    
026        private final static long multiplier = 0x5DEECE66DL;
027        private final static long addend = 0xBL;
028        private final static long mask = (1L << 48) - 1;
029    
030        private static final AtomicLong seq = new AtomicLong(-715159705);
031        private long seed;
032    
033        public AtomicSimpleRandom(long s) {
034            this.seed = s;
035        }
036    
037        public AtomicSimpleRandom() {
038            this.seed = System.nanoTime() + seq.getAndAdd(129);
039        }
040    
041        public void setSeed(long s) {
042            seed = s;
043        }
044    
045        public int nextInt() {
046            return next();
047        }
048    
049        public int next() {
050            long nextseed = (seed * multiplier + addend) & mask;
051            this.seed = nextseed;
052            return ((int) (nextseed >>> 17)) & 0x7FFFFFFF;
053        }
054    
055        public int next(int bits) {
056            long nextseed = (seed * multiplier + addend) & mask;
057            this.seed = nextseed;
058            return (int) (nextseed >>> (48 - bits));
059        }
060    
061    }
