021    package xbird.util.concurrent.counter;
022    
023    import sun.misc.Unsafe;
024    import xbird.util.lang.HashUtils;
025    import xbird.util.lang.UnsafeUtils;
026    
027    /**
028     * 
029     * <DIV lang="en"></DIV>
030     * <DIV lang="ja"></DIV>
031     * 
032     * @author Makoto YUI (yuin405+xbird@gmail.com)
033     */
034    public final class StripeAtomicIntCounter2 implements ICounter {
035        private static final long serialVersionUID = -9202959371917424737L;
036    
037        private static final Unsafe _unsafe = UnsafeUtils.getUnsafe();
038        private static final int _base = _unsafe.arrayBaseOffset(int[].class);
039        private static final int _scale = _unsafe.arrayIndexScale(int[].class);
040        private final int[] _array;
041        private volatile int _macguffin;
042    
043        private final int _size;
044        private final int _mask;
045    
046        private volatile int _fuzzy_sum_cache;
047        private volatile long _fuzzy_time;
048    
049        public StripeAtomicIntCounter2(int nstripe) {
050            assert (nstripe > 0) : nstripe;
051            final int power2 = HashUtils.nextPowerOfTwo(nstripe);
052            this._size = power2;
053            this._mask = power2 - 1;
054    
055            this._array = new int[power2];
056            // must perform at least one volatile write to conform to JMM
057            this._macguffin = 0;
058        }
059    
060        public int get() {
061            int sum = getVolatile(0); // force volatile read
062            final int[] ary = _array;
063            for(int i = 1; i < _size; i++) {
064                sum += ary[i];
065            }
066            return sum;
067        }
068    
069        public int estimateGet() {
070            if(_size < 64) {
071                return get();
072            }
073            final long millis = System.currentTimeMillis();
074            if(_fuzzy_time != millis) {
075                _fuzzy_sum_cache = get();
076                _fuzzy_time = millis; // Indicate freshness of cached value
077            }
078            return _fuzzy_sum_cache;
079        }
080    
081        public void add(final int x) {
082            int idx = HashUtils.hash(Thread.currentThread(), _size - 1);
083            addAndGet(idx, x);
084        }
085    
086        public void add2(final int x) {
087            int hash = HashUtils.hash(Thread.currentThread());
088            int idx = hash & _mask;
089            addAndGet(idx, x);
090        }
091    
092        public int getAndAdd(final int x) {
093            int l = get();
094            add(x);
095            return l;
096        }
097    
098        public int addAndGet(final int x) {
099            add(x);
100            return get();
101        }
102    
103        public int incrementAndGet() {
104            return addAndGet(1);
105        }
106    
107        public int decrementAndGet() {
108            return addAndGet(-1);
109        }
110    
111        public int getAndIncrement() {
112            return getAndAdd(1);
113        }
114    
115        public int getAndDecrement() {
116            return getAndAdd(-1);
117        }
118    
119        private int addAndGet(final int i, final int delta) {
120            while(true) {
121                int current = getVolatile(i);
122                int next = current + delta;
123                if(compareAndSet(i, current, next)) {
124                    return next;
125                }
126            }
127        }
128    
129        private int getVolatile(final int i) {
130            return _array[i + _macguffin];
131        }
132    
133        private boolean compareAndSet(final int i, final int expect, final int update) {
134            return _unsafe.compareAndSwapInt(_array, rawIndex(i), expect, update);
135        }
136    
137        private long rawIndex(final int i) {
138            assert (i >= 0 && i < _array.length) : "Illegal index: " + i;
139            return _base + i * _scale;
140        }
141    
142    }
