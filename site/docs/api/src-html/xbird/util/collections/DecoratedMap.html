021    package xbird.util.collections;
022    
023    import java.util.*;
024    
025    /**
026     * 
027     * <DIV lang="en"></DIV>
028     * <DIV lang="ja"></DIV>
029     * 
030     * @author Makoto YUI (yuin405+xbird@gmail.com)
031     */
032    public class DecoratedMap<K, V> implements Map<K, V> {
033    
034        private final Map<K, V> _entries;
035        private final Map<K, V> _decorated;
036    
037        public DecoratedMap(Map<K, V> decorated) {
038            this(decorated, new HashMap<K, V>());
039        }
040    
041        public DecoratedMap(Map<K, V> decorated, Map<K, V> entries) {
042            if (decorated == null) {
043                throw new IllegalArgumentException("argument MUST not be null.");
044            }
045            this._entries = entries;
046            this._decorated = decorated;
047        }
048    
049        public int size() {
050            return _entries.size() + _decorated.size();
051        }
052    
053        public boolean isEmpty() {
054            return _entries.isEmpty() && _decorated.isEmpty();
055        }
056    
057        public boolean containsKey(Object key) {
058            return _entries.containsKey(key) || _decorated.containsKey(key);
059        }
060    
061        public boolean containsValue(Object value) {
062            return _entries.containsValue(value) || _decorated.containsValue(value);
063        }
064    
065        public V get(Object key) {
066            V v = _decorated.get(key);
067            if (v == null) {
068                return _entries.get(key);
069            } else {
070                return v;
071            }
072        }
073    
074        public V put(K key, V value) {
075            if (_decorated.containsKey(key)) {
076                return _decorated.put(key, value);
077            } else {
078                return _entries.put(key, value);
079            }
080        }
081    
082        public V remove(Object key) {
083            V v = _decorated.remove(key);
084            if (v == null) {
085                return _entries.get(key);
086            } else {
087                return v;
088            }
089        }
090    
091        public void putAll(Map<? extends K, ? extends V> t) {
092            for(Map.Entry<? extends K, ? extends V> e : t.entrySet()) {
093                put(e.getKey(), e.getValue());
094            }
095        }
096    
097        public void clear() {
098            _entries.clear();
099            _decorated.clear();
100        }
101    
102        public Set<K> keySet() {
103            Set<K> s = new HashSet<K>();
104            s.addAll(_entries.keySet());
105            s.addAll(_decorated.keySet());
106            return s;
107        }
108    
109        public Collection<V> values() {
110            List<V> v = new LinkedList<V>();
111            v.addAll(_entries.values());
112            v.addAll(_decorated.values());
113            return v;
114        }
115    
116        public Set<Map.Entry<K, V>> entrySet() {
117            Set<Map.Entry<K, V>> s = new HashSet<Map.Entry<K, V>>();
118            s.addAll(_entries.entrySet());
119            s.addAll(_decorated.entrySet());
120            return s;
121        }
122    
123    }
