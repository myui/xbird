021    package xbird.util.collections;
022    
023    import java.io.Serializable;
024    import java.util.Arrays;
025    
026    /**
027     * Dynamic Integer array.
028     * <DIV lang="en">
029     * This class does not provide strict equality features.
030     * </DIV>
031     * <DIV lang="ja">
032     * </DIV>
033     * 
034     * @author Makoto YUI <yuin@bb.din.or.jp>
035     */
036    public final class CharArrayList implements Serializable {
037        private static final long serialVersionUID = 12356521778076725L;
038    
039        public static final int DEFAULT_CAPACITY = 12;
040    
041        /** array entity */
042        private char[] data;
043        private int used;
044    
045        public CharArrayList() {
046            this(DEFAULT_CAPACITY);
047        }
048    
049        public CharArrayList(int size) {
050            this.data = new char[size];
051            this.used = 0;
052        }
053    
054        public CharArrayList(char[] initValues) {
055            this(initValues.length);
056            add(initValues);
057        }
058    
059        public void add(char value) {
060            if(used >= data.length) {
061                expand(used + 1);
062            }
063            data[used++] = value;
064        }
065    
066        public void add(char[] values) {
067            final int needs = used + values.length;
068            if(needs >= data.length) {
069                expand(needs);
070            }
071            System.arraycopy(values, 0, data, used, values.length);
072            this.used = needs;
073        }
074    
075        /**
076         * dynamic expansion.
077         */
078        private void expand(int max) {
079            while(data.length < max) {
080                final int len = data.length;
081                char[] newArray = new char[len * 2];
082                System.arraycopy(data, 0, newArray, 0, len);
083                this.data = newArray;
084            }
085        }
086    
087        public char remove() {
088            return data[--used];
089        }
090    
091        public char remove(int index) {
092            final char ret;
093            if(index > used) {
094                throw new IndexOutOfBoundsException();
095            } else if(index == used) {
096                ret = data[--used];
097            } else { // index < used
098                // removed value
099                ret = data[index];
100                final char[] newarray = new char[--used];
101                // prefix
102                System.arraycopy(data, 0, newarray, 0, index - 1);
103                // appendix
104                System.arraycopy(data, index + 1, newarray, index, used - index);
105                // set fields.
106                this.data = newarray;
107            }
108            return ret;
109        }
110    
111        public void set(int index, char value) {
112            if(index > used) {
113                throw new IllegalArgumentException("Index MUST be less than \"size()\".");
114            } else if(index == used) {
115                ++used;
116            }
117            data[index] = value;
118        }
119    
120        public char get(int index) {
121            if(index >= used) {
122                throw new IndexOutOfBoundsException();
123            }
124            return data[index];
125        }
126    
127        public int indexOf(char key) {
128            return Arrays.binarySearch(data, key);
129        }
130    
131        public int size() {
132            return used;
133        }
134    
135        public void clear() {
136            used = 0;
137        }
138    
139        public char[] toArray() {
140            final char[] newArray = new char[used];
141            System.arraycopy(data, 0, newArray, 0, used);
142            return newArray;
143        }
144    
145        public char[] array() {
146            return data;
147        }
148    
149        @Override
150        public String toString() {
151            final StringBuilder buf = new StringBuilder();
152            buf.append('[');
153            for(int i = 0; i < used; i++) {
154                if(i != 0) {
155                    buf.append(", ");
156                }
157                buf.append(data[i]);
158            }
159            buf.append(']');
160            return buf.toString();
161        }
162    }
