038    public abstract class Request implements Externalizable {
039    
040        public static final short NORMAL_PRIORITY = 0;
041        public static final short MAX_PRIORITY = Short.MAX_VALUE;
042        public static final short MIN_PRIORITY = Short.MIN_VALUE;
043        /** wait() equals to wait(0) actually, so NO_TIMEOUT is 0. */
044        public static final long NO_TIMEOUT = 0;
045    
046        /** implicit return type */
047        protected/* final */ReturnType _retType;
048        protected String _id;
049        /** by the default, (synchronous) response */
050        protected ReplyPattern _replyPattern = ReplyPattern.RESPONSE;
051        protected ResultHandler _resultHandler = null;
052    
053        // auxiliary properties
054        protected short _priority = NORMAL_PRIORITY;
055        protected long _timeout = NO_TIMEOUT;
056    
057        // latency calculation stuff
058        private long _invokedTimestamp = -1L;
059    
060        public Request() {//Externalizable
061            this(ReturnType.AUTO);
062        }
063    
064        protected Request(ReturnType retType) {
065            if(retType == null) {
066                throw new IllegalArgumentException();
067            }
068            this._retType = retType;
069            this._id = getSignature() + "@" + UUID.randomUUID().toString();
070        }
071    
072        protected Request(String id) {
073            this(id, ReturnType.AUTO);
074        }
075    
076        protected Request(String id, ReturnType retType) {
077            if(id == null) {
078                throw new IllegalArgumentException();
079            }
080            if(retType == null) {
081                throw new IllegalArgumentException();
082            }
083            this._retType = retType;
084            this._id = id;
085        }

087        public abstract Signature getSignature();
088    
089        public ReturnType getReturnType() {
090            return _retType;
091        }
092

094            return _replyPattern;
095        }
096    
097        public void setReplyPattern(ReplyPattern pattern) {
098            this._replyPattern = pattern;
099        }
100    
101        public ResultHandler getResultHandler() {
102            return _resultHandler;
103        }
104    
105        public void setResultHandler(ResultHandler resultHandler) {
106            this._resultHandler = resultHandler;
107        }
108    
109        public String getIdentifier() {
110            return _id;
111        }
112    
113        public void setIdentifier(String id) {
114            this._id = id;
115        }
116    
117        public short getPriority() {
118            return _priority;
119        }
120    
121        public void setPriority(short priority) {
122            this._priority = priority;
123        }
124

126            return _timeout;
127        }
128    
129        public void setTimeout(long timeout) {
130            if(timeout < 0) {
131                throw new IllegalArgumentException("Illegal timeout: " + timeout);
132            }
133            this._timeout = timeout;
134        }
135    
136        public void setInvoked(long timestamp) {
137            this._invokedTimestamp = timestamp;
138        }
139    
140        public long getInvoked() {
141            return _invokedTimestamp;
142        }
143

146            return getIdentifier();
147        }

151        public enum Signature {
152    
153            COMMAND("Command"), QUERY("Query"), PREPARED_QUERY("PreparedQuery"), COMPILE("Compile");
154    
155            final String name;
156    
157            Signature(String name) {
158                this.name = name;
159            }
160    
161            @Override
162            public String toString() {
163                return name;
164            }
165        }

167        public enum ReturnType {
168            AUTO, STRING, BOOLEAN, /* not recommended */SEQUENCE, MARSHALLED_SEQUENCE, REMOTE_SEQUENCE, ASYNC_REMOTE_SEQUENCE, THROTTLED_ASYNC_REMOTE_SEQUENCE, REDIRECTABLE_MARSHALLED_SEQUENCE, REMOTE_PADED_SEQUENCE;
169    
170            public boolean isRemoteSequnece() {
171                return this == ASYNC_REMOTE_SEQUENCE || this == REMOTE_SEQUENCE
172                        || this == THROTTLED_ASYNC_REMOTE_SEQUENCE || this == REMOTE_PADED_SEQUENCE;
173            }
174    
175            public static ReturnType readFrom(final ObjectInput in) throws IOException {
176                final int type = in.readByte();
177                switch(type) {
178                    case 0:
179                        return AUTO;
180                    case 1:
181                        return STRING;
182                    case 2:
183                        return BOOLEAN;
184                    case 3:
185                        return SEQUENCE;
186                    case 4:
187                        return REMOTE_SEQUENCE;
188                    case 5:
189                        return ASYNC_REMOTE_SEQUENCE;
190                    case 6:
191                        return THROTTLED_ASYNC_REMOTE_SEQUENCE;
192                    case 7:
193                        return MARSHALLED_SEQUENCE;
194                    case 8:
195                        return REDIRECTABLE_MARSHALLED_SEQUENCE;
196                    case 9:
197                        return REMOTE_PADED_SEQUENCE;
198                    default:
199                        throw new IllegalStateException("Unexpected type: " + type);
200                }
201            }
202    
203            public void writeTo(final ObjectOutput out) throws IOException {
204                switch(this) {
205                    case AUTO:
206                        out.writeByte(0);
207                        break;
208                    case STRING:
209                        out.writeByte(1);
210                        break;
211                    case BOOLEAN:
212                        out.writeByte(2);
213                        break;
214                    case SEQUENCE:
215                        out.writeByte(3);
216                        break;
217                    case REMOTE_SEQUENCE:
218                        out.writeByte(4);
219                        break;
220                    case ASYNC_REMOTE_SEQUENCE:
221                        out.writeByte(5);
222                        break;
223                    case THROTTLED_ASYNC_REMOTE_SEQUENCE:
224                        out.writeByte(6);
225                        break;
226                    case MARSHALLED_SEQUENCE:
227                        out.writeByte(7);
228                        break;
229                    case REDIRECTABLE_MARSHALLED_SEQUENCE:
230                        out.writeByte(8);
231                        break;
232                    case REMOTE_PADED_SEQUENCE:
233                        out.writeByte(9);
234                        break;
235                    default:
236                        throw new IllegalStateException("Unexpected: " + this);
237                }
238            }
239        }
240    
241        public static final ReturnType resolveReturnType(final String type) {
242            if("AUTO".equalsIgnoreCase(type)) {
243                return ReturnType.AUTO;
244            } else if("THROTTLED_ASYNC_REMOTE_SEQUENCE".equalsIgnoreCase(type)) {
245                return ReturnType.THROTTLED_ASYNC_REMOTE_SEQUENCE;
246            } else if("MARSHALLED_SEQUENCE".equalsIgnoreCase(type)) {
247                return ReturnType.MARSHALLED_SEQUENCE;
248            } else if("REDIRECTABLE_MARSHALLED_SEQUENCE".equalsIgnoreCase(type)) {
249                return ReturnType.REDIRECTABLE_MARSHALLED_SEQUENCE;
250            } else if("REMOTE_PAGED_SEQUENCE".equalsIgnoreCase(type)) {
251                return ReturnType.REMOTE_PADED_SEQUENCE;
252            } else if("ASYNC_REMOTE_SEQUENCE".equalsIgnoreCase(type)) {
253                return ReturnType.ASYNC_REMOTE_SEQUENCE;
254            } else if("REMOTE_SEQUENCE".equalsIgnoreCase(type)) {
255                return ReturnType.REMOTE_SEQUENCE;
256            } else if("SEQUENCE".equalsIgnoreCase(type)) {
257                return ReturnType.SEQUENCE;
258            } else if("STRING".equalsIgnoreCase(type)) {
259                return ReturnType.STRING;
260            } else if("BOOLEAN".equalsIgnoreCase(type)) {
261                return ReturnType.BOOLEAN;
262            } else {
263                throw new IllegalArgumentException("Unsupported type: " + type);
264            }
265        }

267        /** 
268         * The default reply pattern is (synchronous) Response.
269         * (Asynchronous) Callback reply pattern is not recommended, since
270         * a client takes an RMI-connection until callback is invocated
271         * and callback does not work if client is behind a firewall (includes Windows Firewall).
272         * Note that Poll reply pattern spends server's cache-resource until 
273         * poll-request is invocated. Pending responses are flushed to disk 
274         * when the cache is overflowed.
275         * The cache size on main-memory depends on the ehcache configuration.
276         * 
277         * @link http://java.sun.com/developer/onlineTraining/rmi/RMI.html#FirewallIssues
278         * @link http://java.sun.com/j2se/1.5.0/docs/guide/rmi/spec/rmi-arch6.html
279         */
280        public enum ReplyPattern {
281            RESPONSE, CALLBACK, POLL;
282    
283            public static ReplyPattern readFrom(final ObjectInput in) throws IOException {
284                final int ptn = in.readByte();
285                switch(ptn) {
286                    case 1:
287                        return RESPONSE;
288                    case 2:
289                        return CALLBACK;
290                    case 3:
291                        return POLL;
292                    default:
293                        throw new IllegalStateException("Illegal pattern: " + ptn);
294                }
295            }
296    
297            public void writeTo(final ObjectOutput out) throws IOException {
298                switch(this) {
299                    case RESPONSE:
300                        out.writeByte(1);
301                        break;
302                    case CALLBACK:
303                        out.writeByte(2);
304                        break;
305                    case POLL:
306                        out.writeByte(3);
307                        break;
308                    default:
309                        throw new IllegalStateException("Unexpected: " + this);
310                }
311            }
312        }
313    
314        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
315            this._retType = ReturnType.readFrom(in);
316            this._id = ObjectUtils.readString(in);
317            final int auxiliary = in.readByte();
318            if(auxiliary != 0) {
319                if((auxiliary & 1) == 1) {
320                    this._replyPattern = ReplyPattern.readFrom(in);
321                }
322                if((auxiliary & 2) == 2) {
323                    this._resultHandler = (ResultHandler) in.readObject();
324                }
325                if((auxiliary & 4) == 4) {
326                    this._priority = in.readShort();
327                }
328                if((auxiliary & 8) == 8) {
329                    this._timeout = in.readLong();
330                }
331            }
332            this._invokedTimestamp = in.readLong();
333        }
334    
335        public void writeExternal(ObjectOutput out) throws IOException {
336            _retType.writeTo(out); // #1
337            assert (_id != null);
338            ObjectUtils.writeString(out, _id); // #2
339            int auxiliary = 0;
340            final ReplyPattern replyPattern = _replyPattern;
341            if(replyPattern != ReplyPattern.RESPONSE) {
342                auxiliary |= 1;
343            }
344            final ResultHandler handler = _resultHandler;
345            if(handler != null) {
346                auxiliary |= 2;
347            }
348            final short priority = _priority;
349            if(priority != NORMAL_PRIORITY) {
350                auxiliary |= 4;
351            }
352            final long timeout = _timeout;
353            if(timeout != 0) {
354                auxiliary |= 8;
355            }
356            out.writeByte(auxiliary); // #3 controls
357            if(auxiliary != 0) {
358                if(replyPattern != ReplyPattern.RESPONSE) {
359                    replyPattern.writeTo(out);
360                }
361                if(handler != null) {
362                    out.writeObject(handler);
363                }
364                if(priority != NORMAL_PRIORITY) {
365                    out.writeShort(priority);
366                }
367                if(timeout != 0) {
368                    out.writeLong(timeout);
369                }
370            }
371            out.writeLong(_invokedTimestamp);
372        }
