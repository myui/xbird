050    public final class PreparedQueryProcessor extends QueryProcessor {
051        private static final Log LOG = LogFactory.getLog(PreparedQueryProcessor.class);
052    
053        public PreparedQueryProcessor(ResponseListener handler) {
054            super(handler);
055        }
056    
057        @Override
058        public Signature associatedWith() {
059            return Signature.PREPARED_QUERY;
060        }
061    
062        @Override
063        public void fire(RequestContext rc) throws RemoteException {
064            rc.setFired(System.currentTimeMillis());
065    
066            final Request request = rc.getRequest();
067            final Signature rsig = request.getSignature();
068            if(rsig != Signature.PREPARED_QUERY) {
069                throw new IllegalStateException("Illegal command is passed to PreparedQueryProcessor: "
070                        + rsig);
071            }
072            final PreparedQueryRequest queryRequest = (PreparedQueryRequest) request;
073            final XQExpression queryExpr = queryRequest.getCompiledExpression();
074            if(queryExpr == null) {
075                throw new IllegalStateException("query was null for: " + queryRequest);
076            }
077    
078            final ThreadedVariableCollector collector = new ThreadedVariableCollector();
079            try {
080                collector.visit(queryExpr, null);
081            } catch (XQueryException e) {
082                throw new RemoteException("failed runnning ThreadVariables", e);
083            }
084            final DynamicContext dynEnv = new DynamicContext(new StaticContext());
085            dynEnv.setQueryExpression(queryExpr);
086            collector.invokeThreadedVariables(dynEnv); // TODO REVIEWME cancelling
087    
088            final Sequence<? extends Item> resultSeq;
089            _runningThreads.put(rc, Thread.currentThread());
090            try {
091                resultSeq = queryExpr.eval(null, dynEnv);
092            } catch (XQueryException e) {
093                LOG.error("execute failed: \n" + queryExpr, e);
094                rc.setFault(e);
095                _resHandler.onResponse(rc);
096                return;
097            } finally {
098                _runningThreads.remove(rc);
099            }
100    
101            final Serializable result = wrapResult(resultSeq, queryExpr, queryRequest, dynEnv);
102            rc.setResult(result);
103    
104            try {
105                _resHandler.onResponse(rc);
106            } catch (RemoteException re) {
107                LOG.error("The respond for the request '" + rc.getRequest().getIdentifier()
108                        + "' is failed", re);
109            }
110        }
111    
112        private static final class ThreadedVariableCollector extends AbstractXQueryParserVisitor {
113    
114            private final List<ThreadedVariable> _threadedVars = new ArrayList<ThreadedVariable>(4);
115    
116            public ThreadedVariableCollector() {
117                super();
118            }
119    
120            public void invokeThreadedVariables(final DynamicContext dynEnv) {
121                for(ThreadedVariable v : _threadedVars) {
122                    v.setDynamicContext(dynEnv);
123                    final String threadName = "ThreadedVariable#" + v.getName();
124                    final Thread task = new Thread(v, threadName);
125                    task.setDaemon(true);
126                    task.start();
127                }
128            }
129    
130            @Override
131            public XQExpression visit(Variable variable, XQueryContext ctxt) throws XQueryException {
132                if(variable instanceof ThreadedVariable) {
133                    assert (!_threadedVars.contains(variable));
134                    ThreadedVariable thVar = (ThreadedVariable) variable;
135                    _threadedVars.add(thVar);
136                    return variable;
137                }
138                return super.visit(variable, ctxt);
139            }
140    
141        }
142    }
