058    public class XQueryServlet extends HttpServlet {
059        private static final long serialVersionUID = -4576560807978002697L;
060    
061        public static final String XQSP_NSURI = "xbird://xqsp";
062    
063        private final ReadWriteLock _lock = new ReentrantReadWriteLock();
064        private final Cache<String, CachedQuery> _caches = CacheFactory.createCache(XQueryServlet.class.getName());
065    
066        public XQueryServlet() {
067            super();
068        }
069    
070        @Override
071        public void init() throws ServletException {
072            super.init();
073        }
074    
075        @Override
076        protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException,
077                IOException {
078            process(req, resp);
079        }
080    
081        @Override
082        protected void doPost(HttpServletRequest req, HttpServletResponse resp)
083                throws ServletException, IOException {
084            process(req, resp);
085        }
086    
087        protected void process(HttpServletRequest req, HttpServletResponse resp) throws IOException {
088            resp.setCharacterEncoding("UTF-8");
089            resp.setContentType("text/html");
090            PrintWriter out = resp.getWriter();
091            String queryPath = req.getServletPath();
092            final CachedQuery loaded;
093            try {
094                loaded = loadQuery(queryPath);
095            } catch (XQueryException e) {
096                reportError("Query loading failed: " + queryPath, e, out);
097                return;
098            } catch (CacheException e) {
099                reportError("Caching failed: " + queryPath, e, out);
100                return;
101            }
102            XQueryModule loadedModule = loaded.queryObject;
103            XQExpression body = loadedModule.getExpression();
104            if(body == null) {
105                return;
106            }
107            // set parameters
108            Enumeration<String> paramNames = req.getParameterNames();
109            while(paramNames.hasMoreElements()) {
110                String name = paramNames.nextElement();
111                String value = req.getParameter(name);
112                QualifiedName qname = QNameUtil.parse(name, XQSP_NSURI);
113                Variable var = new GlobalVariable(qname, null);
114                var.setResult(XString.valueOf(value));
115                try {
116                    loadedModule.putVariable(qname, var);
117                } catch (XQueryException e) {
118                    reportError("Setting parameter failed: " + qname, e, out);
119                    return;
120                }
121            }
122            // execute
123            final Sequence<? extends Item> result;
124            try {
125                result = body.eval(null, new DynamicContext(loaded.staticEnv));
126            } catch (XQueryException e) {
127                reportError("Execution failed: " + queryPath, e, out);
128                return;
129            }
130            // serialize        
131            SAXWriter writer = new SAXWriter(out);
132            Serializer ser = new SAXSerializer(writer);
133            try {
134                ser.emit(result);
135            } catch (XQueryException e) {
136                reportError("Serialization failed: " + queryPath, e, out);
137                return;
138            }
139        }
140    
141        private CachedQuery loadQuery(String path) throws IOException, XQueryException, CacheException {
142            URL url = getServletContext().getResource(path);
143            assert (url != null);
144            long lastModified = url.openConnection().getLastModified();
145            _lock.readLock().lock();
146            CachedQuery cached = _caches.get(path);
147            if(cached == null || cached.loadTimeStamp < lastModified) {
148                if(cached == null) {
149                    cached = new CachedQuery();
150                }
151                // parse XQuery expression
152                InputStream is = url.openStream();
153                XQueryParser parser = new XQueryParser(is);
154                StaticContext staticEnv = parser.getStaticContext();
155                staticEnv.setBaseURI(path);
156                XQueryModule module = parser.parse();
157                _lock.readLock().unlock();
158                _lock.writeLock().lock();
159                // set query cache
160                cached.queryObject = module;
161                cached.staticEnv = staticEnv;
162                cached.loadTimeStamp = System.currentTimeMillis();
163                _caches.put(path, cached);
164                _lock.writeLock().unlock();
165                _lock.readLock().lock();
166                // static analysis
167                module.staticAnalysis(staticEnv);
168            }
169            _lock.readLock().unlock();
170            return cached;
171        }
172    
173        private static void reportError(String title, Exception e, PrintWriter out) {
174            out.println("<h2>" + title + "</h2>");
175            out.println("<pre>");
176            String trace = PrintUtils.prettyPrintStackTrace(e);
177            out.println(trace);
178            out.println("</pre>");
179        }
180    
181        private static final class CachedQuery {
182            private XQueryModule queryObject;
183            private StaticContext staticEnv;
184            private long loadTimeStamp;
185    
186            private CachedQuery() {}
187    
188        }
