021    package xbird.xquery.dm.value.sequence;
022    
023    import xbird.util.io.IOUtils;
024    import xbird.util.iterator.MultiplexingIterator;
025    import xbird.xquery.XQueryException;
026    import xbird.xquery.dm.value.AbstractSequence;
027    import xbird.xquery.dm.value.Item;
028    import xbird.xquery.dm.value.Sequence;
029    import xbird.xquery.meta.DynamicContext;
030    import xbird.xquery.meta.Focus;
031    import xbird.xquery.meta.IFocus;
032    import xbird.xquery.type.SequenceType;
033    import xbird.xquery.type.Type;
034    
035    /**
036     * 
037     * <DIV lang="en"></DIV>
038     * <DIV lang="ja"></DIV>
039     * 
040     * @author Makoto YUI (yuin405+xbird@gmail.com)
041     */
042    public final class PipelinedSequence extends AbstractSequence {
043        private static final long serialVersionUID = -4626518122143252247L;
044    
045        private final Sequence[] _sequences;
046    
047        public PipelinedSequence(DynamicContext dynEnv, Sequence... seqs) {
048            super(dynEnv);
049            if(seqs == null) {
050                throw new IllegalArgumentException();
051            }
052            this._sequences = seqs;
053        }
054    
055        @SuppressWarnings("unchecked")
056        @Override
057        public IFocus iterator() {
058            final int seqlen = _sequences.length;
059            final IFocus[] focuz = new IFocus[seqlen];
060            for(int i = 0; i < seqlen; i++) {
061                focuz[i] = _sequences[i].iterator();
062            }
063            MultiplexingIterator<Item> itor = new MultiplexingIterator<Item>(focuz);
064            return new Focus<Item>(this, itor, _dynEnv);
065        }
066    
067        @SuppressWarnings("unchecked")
068        public boolean next(IFocus focus) throws XQueryException {
069            final MultiplexingIterator<Item> itor = (MultiplexingIterator<Item>) focus.getBaseFocus();
070            while(itor.hasNext()) {
071                Item it = itor.next();
072                focus.setContextItem(it);
073                return true;
074            }
075            focus.setReachedEnd(true);
076            IOUtils.closeQuietly(itor);
077            return false;
078        }
079    
080        public Type getType() {
081            return SequenceType.ANY_ITEMS;
082        }
083    
084    }
