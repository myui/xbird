040    /**
041     * fn:index-of.
042     * <DIV lang="en">
043     * Returns a sequence of positive integers giving the positions 
044     * within the sequence $seqParam of items that are equal to $srchParam.
045     * <ul>
046     * <li>fn:index-of($seqParam as xdt:anyAtomicType*, $srchParam as xdt:anyAtomicType) as xs:integer*</li>
047     * <li>fn:index-of($seqParam as xdt:anyAtomicType*, $srchParam as xdt:anyAtomicType, $collation as xs:string) as xs:integer*</li>
048     * </ul>
049     * </DIV>
050     * <DIV lang="ja"></DIV>
051     * 
052     * @author Makoto YUI (yuin405+xbird@gmail.com)
053     * @link http://www.w3.org/TR/xquery-operators/#func-index-of
054     */
055    public final class IndexOf extends BuiltInFunction {
056        private static final long serialVersionUID = 1498600419388016663L;
057        public static final String SYMBOL = "fn:index-of";
058    
059        public IndexOf() {
060            super(SYMBOL, TypeRegistry.safeGet("xs:integer*"));
061        }
062    
063        protected FunctionSignature[] signatures() {
064            final FunctionSignature[] s = new FunctionSignature[2];
065            final Type anyatom = TypeRegistry.safeGet("xs:anyAtomicType*");
066            s[0] = new FunctionSignature(getName(), new Type[] { anyatom, AtomicType.ANY_ATOMIC_TYPE });
067            s[1] = new FunctionSignature(getName(), new Type[] { anyatom, AtomicType.ANY_ATOMIC_TYPE,
068                    StringType.STRING });
069            return s;
070        }
071    
072        /**
073         * Note:
074         *  The first item in a sequence is at position 1, not position 0.
075         *  The result sequence is in ascending numeric order.
076         */
077        public Sequence eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
078                throws XQueryException {
079            final Sequence<? extends Item> src = argv.getItem(0).atomize(dynEnv);
080            final Item target = argv.getItem(1);
081            final AtomicValue searchItem = (AtomicValue) target;
082            final Item coll = (argv.size() == 3) ? argv.getItem(2) : null;
083            return new IndexOfEmuration(src, searchItem, (coll != null) ? coll.stringValue() : null, dynEnv);
084        }
085    
086        private static final class IndexOfEmuration extends ProxySequence {
087            private static final long serialVersionUID = 3824666835912342600L;
088    
089            private final AtomicValue _searchItem;
090            private final String _collation;
091            /** The collation is used when string comparison is required. */
092            private transient Collator _collator = null;
093            private String _searchItemStr;
094    
095            public IndexOfEmuration(Sequence<? extends Item> src, AtomicValue searchItem, String collation, DynamicContext dynEnv) {
096                super(src, dynEnv);
097                assert (searchItem != null);
098                this._searchItem = searchItem;
099                this._collation = collation;
100                if(TypeUtil.subtypeOf(searchItem.getType(), StringType.STRING)) {
101                    this._searchItemStr = searchItem.stringValue();
102                    if(collation != null) {
103                        this._collator = CollationUtils.resolve(collation, dynEnv.getStaticContext());
104                    }
105                }
106            }
107    
108            public boolean next(IFocus focus) throws XQueryException {
109                Iterator<? extends Item> srcItor = focus.getBaseFocus();
110                while(srcItor.hasNext()) {
111                    Item srcItem = srcItor.next();
112                    assert (srcItem != null);
113                    if(found(srcItem)) {
114                        focus.setContextItem(srcItem);
115                        return true;
116                    }
117                }
118                return false;
119            }
120    
121            private boolean found(Item srcItem) {
122                final int cmp;
123                // TODO when string comparison is required?
124                if(_searchItemStr != null && _collator != null) {
125                    cmp = _collator.compare(srcItem, _searchItemStr);
126                } else {
127                    cmp = srcItem.compareTo(_searchItem);
128                }
129                return cmp == 0;
130            }
131    
132            @Override
133            public Type getType() {
134                return TypeRegistry.safeGet("xs:integer*");
135            }
136    
137            private Object readResolve() throws ObjectStreamException {
138                if(_collator == null && _collation != null) {
139                    assert (_dynEnv != null);
140                    this._collator = CollationUtils.resolve(_collation, _dynEnv.getStaticContext());
141                }
142                return this;
143            }
144    
145        }
