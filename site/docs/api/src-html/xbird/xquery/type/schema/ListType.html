021    package xbird.xquery.type.schema;
022    
023    import java.util.*;
024    
025    import xbird.xquery.type.*;
026    
027    /**
028     * 
029     * <DIV lang="en"></DIV>
030     * <DIV lang="ja"></DIV>
031     * 
032     * @author Makoto YUI (yuin405+xbird@gmail.com)
033     */
034    public final class ListType extends Type implements SchemaType {
035        private static final long serialVersionUID = -2479802443238568170L;
036    
037        private final List<Type> _list;
038        private final Set<Type> _set;
039    
040        private boolean hasNoneType = false;
041    
042        public ListType(List<Type> t) {
043            this._list = t;
044            this._set = new HashSet<Type>(8);
045        }
046    
047        public ListType(int size) {
048            this(new ArrayList<Type>(size));
049        }
050    
051        @Deprecated
052        public ListType(Type... types) {
053            this(new ArrayList<Type>(types.length));
054            for(Type t : types) {
055                if(t == Type.NONE) {
056                    this.hasNoneType = true;
057                }
058                _list.add(t);
059                _set.add(t);
060            }
061        }
062    
063        public void add(Type t) {
064            if(t == null) {
065                throw new IllegalArgumentException();
066            }
067            if(!hasNoneType) {
068                if(t == Type.NONE) {
069                    this.hasNoneType = true;
070                }
071                _list.add(t);
072                _set.add(t);
073            }
074        }
075    
076        public Type get(int i) {
077            if(i >= size()) {
078                throw new IndexOutOfBoundsException("Out of index: " + i);
079            }
080            return _list.get(i);
081        }
082    
083        public int size() {
084            return _list.size();
085        }
086    
087        public List<Type> asList() {
088            return _list;
089        }
090    
091        public Occurrence quantifier() {
092            final int size = _list.size();
093            if(size == 0) {
094                return Occurrence.OCC_ZERO;
095            } else if(size == 1) {
096                return Occurrence.OCC_EXACTLY_ONE;
097            } else {
098                return Occurrence.OCC_ONE_OR_MORE;
099            }
100        }
101    
102        public boolean accepts(Type expected) {
103            if(expected instanceof ListType) {
104                ListType elist = (ListType) expected;
105                final int maxlen = Math.max(_list.size(), elist.size());
106                for(int i = 0; i < maxlen; i++) {
107                    Type t = _list.get(i);
108                    Type e = elist.get(i);
109                    if(!t.accepts(e)) {
110                        return false;
111                    }
112                }
113                return true;
114            } else {
115                if(_list.size() == 1) {
116                    Type t = _list.get(0);
117                    if(t.accepts(expected)) {
118                        return true;
119                    }
120                }
121                return false;
122            }
123        }
124    
125        public Class getJavaObjectType() {
126            return List.class;
127        }
128    
129        private transient Type[] _cachedUniqTypes = null;
130    
131        public Type prime() {
132            if(hasNoneType) {
133                return Type.NONE;
134            }
135            final int uniqTypeLength = _set.size();
136            if(uniqTypeLength == 0) {
137                return SequenceType.EMPTY;
138            } else if(uniqTypeLength == 1) {
139                return _list.get(0);
140            } else {
141                final Type[] uniqTypes;
142                if(_cachedUniqTypes != null && _cachedUniqTypes.length == uniqTypeLength) {
143                    uniqTypes = _cachedUniqTypes;
144                } else {
145                    uniqTypes = new Type[uniqTypeLength];
146                    _set.toArray(uniqTypes);
147                    this._cachedUniqTypes = uniqTypes;
148                }
149                return new ChoiceType(uniqTypes);
150            }
151        }
152    
153        public String toString() {
154            final StringBuilder buf = new StringBuilder(64);
155            buf.append('(');
156            for(int i = 0; i < _list.size(); i++) {
157                if(i != 0) {
158                    buf.append(", ");
159                }
160                Type t = _list.get(i);
161                buf.append(t.toString());
162            }
163            buf.append(')');
164            return buf.toString();
165        }
166    
167    }
