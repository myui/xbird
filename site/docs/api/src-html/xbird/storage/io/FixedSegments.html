021    package xbird.storage.io;
022    
023    import java.io.File;
024    import java.io.IOException;
025    import java.io.RandomAccessFile;
026    
027    /**
028     * Fixed-size pager.
029     * <DIV lang="en"></DIV>
030     * <DIV lang="ja"></DIV>
031     * 
032     * @author Makoto YUI (yuin405+xbird@gmail.com)
033     */
034    public final class FixedSegments implements Segments {
035    
036        private final File file;
037        private final int recordLength;
038        private RandomAccessFile raf;
039        private boolean open = false;
040        private boolean dirty = true;
041    
042        public FixedSegments(final File file, final int recordLength) {
043            this.file = file;
044            this.recordLength = recordLength;
045            try {
046                ensureOpen();
047            } catch (IOException e) {
048                throw new IllegalStateException("loading file failed: " + file.getAbsolutePath(), e);
049            }
050        }
051    
052        private synchronized void ensureOpen() throws IOException {
053            if(open) {
054                return;
055            }
056            if(raf == null) {
057                this.raf = new RandomAccessFile(file, "rw");
058            }
059            this.open = true;
060        }
061    
062        public File getFile() {
063            return file;
064        }
065    
066        public long write(final long idx, final byte[] b) throws IOException {
067            checkRecordLength(b);
068            final long ptr = toPhysicalAddr(idx);
069            ensureOpen();
070            synchronized(raf) {// write to disk
071                raf.seek(ptr);
072                raf.write(b);
073                this.dirty = true;
074            }
075            return ptr;
076        }
077    
078        public long bulkWrite(final long idx, final byte[] b) throws IOException {
079            final long ptr = toPhysicalAddr(idx);
080            ensureOpen();
081            synchronized(raf) {// write to disk
082                raf.seek(ptr);
083                raf.write(b);
084                this.dirty = true;
085            }
086            return ptr;
087        }
088    
089        private final long toPhysicalAddr(final long logicalAddr) {
090            return logicalAddr * recordLength;
091        }
092    
093        private final void checkRecordLength(final byte[] b) {
094            if(b.length != recordLength) {
095                throw new IllegalArgumentException("Invalid Record length: " + b.length
096                        + ", expected: " + recordLength);
097            }
098        }
099    
100        public byte[] read(final long idx) throws IOException {
101            final long ptr = toPhysicalAddr(idx);
102            return directRead(ptr, 1);
103        }
104    
105        public byte[][] readv(long[] idx) throws IOException {
106            final int len = idx.length;
107            final byte[][] pages = new byte[len][];
108            for(int i = 0; i < len; i++) {
109                pages[i] = read(idx[i]);
110            }
111            return pages;
112        }
113    
114        public byte[] bulkRead(final long idx, final int numBlocks) throws IOException {
115            final long ptr = toPhysicalAddr(idx);
116            return directRead(ptr, numBlocks);
117        }
118    
119        public final void bulkRead(final long idx, final byte[] b) throws IOException {
120            ensureOpen();
121            final long addr = toPhysicalAddr(idx);
122            synchronized(raf) {// read from disk
123                raf.seek(addr);
124                raf.read(b);
125            }
126        }
127    
128        private final byte[] directRead(final long addr, final int numBlocks) throws IOException {
129            ensureOpen();
130            final byte[] b = new byte[recordLength * numBlocks];
131            synchronized(raf) {// read from disk
132                raf.seek(addr);
133                raf.read(b);
134            }
135            return b;
136        }
137    
138        public void flush(final boolean close) throws IOException {
139            if(!dirty) {
140                return;
141            }
142            if(!open) {
143                throw new IllegalStateException();
144            }
145            raf.getChannel().force(true);
146            this.dirty = false;
147            if(close) {
148                close();
149            }
150        }
151    
152        public void close() throws IOException {
153            if(raf != null) {
154                raf.close();
155                this.raf = null;
156            }
157            this.open = false;
158        }
159    
160    }
