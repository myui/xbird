100            super(file, pageSize);
101            BTreeFileHeader fh = getFileHeader();

153        @Override
154        public boolean create(boolean close) throws DbException {
155            if(super.create(false)) {
156                // Don't call this.open() as it will try to read rootNode from the disk
157                super.open();
158                // Initialize root node
159                long p = _fileHeader.getRootPage();
160                this._rootInfo = new BTreeRootInfo(p);
161                this._rootNode = new BTreeNode(_rootInfo, getPage(p));
162                _rootNode.ph.setStatus(LEAF);
163                _rootNode.set(new Value[0], new long[0]);
164                try {
165                    _rootNode.write();
166                } catch (IOException e) {
167                    throw new DbException(e);
168                }
169                synchronized(_cache) {
170                    _cache.put(_rootNode.page.getPageNum(), _rootNode);
171                }
172                if(close) {
173                    close();
174                }
175                return true;
176            }
177            return false;
178        }

288        /**
289         * query performs a query against the BTree and performs callback
290         * operations to report the search results.
291         *
292         * @param query The IndexQuery to use
293         * @param callback The callback instance
294         */
295        public void search(IndexQuery query, BTreeCallback callback) throws DbException {
296            if(query == null) {
297                throw new IllegalArgumentException();
298            }
299            BTreeNode root = getRootNode();
300            Value[] keys = query.getOperands();
301            assert (keys.length >= 1) : keys.length;
302            final int op = query.getOperator();
303            try {
304                switch(op) {
305                    case BasicIndexQuery.EQ: {
306                        if(isDuplicateAllowed()) {
307                            BTreeNode left = root.getLeafNode(SearchType.LEFT, keys[0]);
308                            BTreeNode right = root.getLeafNode(SearchType.RIGHT, keys[0]);
309                            scanRange(left, right, query, callback);
310                        } else {
311                            BTreeNode left = root.getLeafNode(SearchType.LEFT, keys[0]);
312                            left.scanLeaf(query, callback, true);
313                        }
314                        break;
315                    }
316                    case BasicIndexQuery.GT:
317                    case BasicIndexQuery.GE: {
318                        BTreeNode right = root.getLeafNode(SearchType.LEFT, keys[keys.length - 1]);
319                        BTreeNode rightmost = root.getLeafNode(SearchType.RIGHT_MOST, null);
320                        scanRange(right, rightmost, query, callback);
321                        break;
322                    }
323                    case BasicIndexQuery.LE:
324                    case BasicIndexQuery.LT: {
325                        BTreeNode left = root.getLeafNode(SearchType.LEFT, keys[0]);
326                        BTreeNode leftmost = root.getLeafNode(SearchType.LEFT_MOST, null);
327                        scanRange(leftmost, left, query, callback);
328                        break;
329                    }
330                    case BasicIndexQuery.NE:
331                    case BasicIndexQuery.NBW:
332                    case BasicIndexQuery.NOT_IN:
333                    case BasicIndexQuery.NOT_START_WITH:
334                    case BasicIndexQuery.NBWX: {
335                        BTreeNode leftmost = root.getLeafNode(SearchType.LEFT_MOST, null);
336                        BTreeNode left = root.getLeafNode(SearchType.LEFT, keys[0]);
337                        BTreeNode rightmost = root.getLeafNode(SearchType.RIGHT_MOST, null);
338                        BTreeNode right = root.getLeafNode(SearchType.RIGHT, keys[keys.length - 1]);
339                        scanRange(leftmost, left, query, callback);
340                        long lp = left.page.getPageNum(), rp = right.page.getPageNum();
341                        if(lp != rp) {
342                            scanRange(right, rightmost, query, callback);
343                        }
344                        break;
345                    }
346                    case BasicIndexQuery.BW:
347                    case BasicIndexQuery.START_WITH:
348                    case BasicIndexQuery.IN:
349                    case BasicIndexQuery.BWX: {
350                        BTreeNode left = root.getLeafNode(SearchType.LEFT, keys[0]);
351                        BTreeNode right = root.getLeafNode(SearchType.RIGHT, keys[keys.length - 1]);
352                        scanRange(left, right, query, callback);
353                        break;
354                    }
355                    default: {
356                        BTreeNode leftmost = root.getLeafNode(SearchType.LEFT_MOST, null);
357                        BTreeNode rightmost = root.getLeafNode(SearchType.RIGHT_MOST, null);
358                        scanRange(leftmost, rightmost, query, callback);
359                        break;
360                    }
361                }
362            } catch (IOException e) {
363                throw new DbException(e);
364            }
365        }

367        private final void scanRange(BTreeNode left, BTreeNode right, IndexQuery query, BTreeCallback callback)
368                throws DbException {
369            final long rightmostPageNum = right.page.getPageNum();
370            if(LOG.isInfoEnabled()) {
371                LOG.info("scan range [" + left.page.getPageNum() + ", " + rightmostPageNum + "] start");
372            }
373            BTreeNode cur = left;
374            int scaned = 0;
375            while(true) {
376                long curPageNum = cur.page.getPageNum();
377                if(curPageNum == rightmostPageNum) {
378                    cur.scanLeaf(query, callback, true);
379                    ++scaned;
380                    break;
381                } else {
382                    cur.scanLeaf(query, callback, scaned == 0);
383                    ++scaned;
384                }
385                long next = cur._next;
386                if(next == curPageNum) {
387                    throw new IllegalStateException("detected a cyclic link at page#" + curPageNum);
388                } else if(next == -1L) {
389                    throw new IllegalStateException("range scan failed... bug?");
390                }
391                cur = getBTreeNode(_rootInfo, next, null);
392            }
393            if(LOG.isInfoEnabled()) {
394                LOG.info("scan range end. total scaned pages: " + scaned);
395            }
396        }

433        private final BTreeNode getBTreeNode(BTreeRootInfo root, long page, BTreeNode parent)
434                throws DbException {
435            BTreeNode node;
436            synchronized(_cache) {
437                node = _cache.get(page);
438                if(node == null) {
439                    node = new BTreeNode(root, getPage(page), parent);
440                    try {
441                        node.read();
442                    } catch (IOException e) {
443                        throw new DbException("failed to read page#" + page, e);
444                    }
445                    if(LOG.isDebugEnabled()) {
446                        LOG.debug("read node page#" + page + ", keys: " + node.keys.length);
447                    }
448                    _cache.put(node.page.getPageNum(), node);
449                } else {
450                    node.parent = parent;
451                }
452            }
453            return node;
454        }

468        public void flush(boolean purge, boolean clear) throws DbException {
469            if(purge) {
470                for(BucketEntry<BTreeNode> e : _cache) {
471                    BTreeNode node = e.getValue();
472                    if(node == null) {
473                        continue;
474                    }
475                    try {
476                        node.write();
477                    } catch (IOException ioe) {
478                        throw new DbException(ioe);
479                    }
480                }
481            }
482            if(clear) {
483                _cache.clear();
484            }
485            super.flush();
486        }

543            private synchronized long addValue(Value value, long pointer) throws IOException,
544                    DbException {
545                if(value == null) {
546                    throw new IllegalArgumentException("Can't add a null Value");
547                }
548                int idx = searchRightmostKey(keys, value, keys.length);
549                switch(ph.getStatus()) {
550                    case BRANCH:
551                        idx = idx < 0 ? -(idx + 1) : idx + 1;
552                        return getChildNode(idx).addValue(value, pointer);
553                    case LEAF:
554                        final boolean found = idx >= 0;
555                        final long oldPtr;
556                        if(found) {
557                            if(!isDuplicateAllowed()) {
558                                throw new BTreeCorruptException("Attempt to add duplicate key to the unique index: "
559                                        + value);
560                            }
561                            oldPtr = ptrs[idx];
562                            value = keys[idx];
563                            idx = idx + 1;
564                        } else {
565                            oldPtr = -1;
566                            idx = -(idx + 1);
567                        }
568                        set(ArrayUtils.<Value> insert(keys, idx, value), ArrayUtils.insert(ptrs, idx, pointer));
569                        incrDataLength(value, pointer);
570    
571                        // Check to see if we've exhausted the block
572                        if(needSplit()) {
573                            split();
574                        }
575                        return oldPtr;
576                    default:
577                        throw new BTreeCorruptException("Invalid Page Type '" + ph.getStatus()
578                                + "' was detected for page#" + page.getPageNum());
579                }
580            }

582            /** search the leftmost key for duplicate allowed index */
583            private int searchLeftmostKey(final Value[] ary, final Value key, final int to) {
584                if(!_fileHeader._duplicateAllowed) {
585                    return Arrays.binarySearch(keys, 0, to, key);
586                }
587                int low = 0;
588                int high = to - 1;
589                while(low <= high) {
590                    int mid = (low + high) >>> 1;
591                    Value midVal = ary[mid];
592                    int cmp = midVal.compareTo(key);
593                    if(cmp < 0) {
594                        low = mid + 1;
595                    } else if(cmp > 0) {
596                        high = mid - 1;
597                    } else {
598                        for(int i = mid - 1; i >= 0; i--) {
599                            Value nxtVal = ary[i];
600                            cmp = midVal.compareTo(nxtVal);
601                            if(cmp != 0) {
602                                break;
603                            }
604                            mid = i;
605                        }
606                        return mid; // key found
607                    }
608                }
609                return -(low + 1); // key not found.
610            }
611    
612            /** search the rightmost key for duplicate allowed index */
613            private int searchRightmostKey(final Value[] ary, final Value key, final int to) {
614                if(!_fileHeader._duplicateAllowed) {
615                    return Arrays.binarySearch(keys, 0, to, key);
616                }
617                int low = 0;
618                int high = to - 1;
619                while(low <= high) {
620                    int mid = (low + high) >>> 1;
621                    Value midVal = ary[mid];
622                    int cmp = midVal.compareTo(key);
623                    if(cmp < 0) {
624                        low = mid + 1;
625                    } else if(cmp > 0) {
626                        high = mid - 1;
627                    } else {
628                        for(int i = mid + 1; i <= high; i++) {
629                            Value nxtVal = ary[i];
630                            cmp = midVal.compareTo(nxtVal);
631                            if(cmp != 0) {
632                                break;
633                            }
634                            mid = i;
635                        }
636                        return mid; // key found
637                    }
638                }
639                return -(low + 1); // key not found.
640            }

642            /** @return pointer of left-most matched item */
643            private synchronized long removeValue(Value value) throws IOException, DbException {
644                int leftIdx = searchLeftmostKey(keys, value, keys.length);
645                switch(ph.getStatus()) {
646                    case BRANCH:
647                        leftIdx = (leftIdx < 0) ? -(leftIdx + 1) : leftIdx + 1;
648                        return getChildNode(leftIdx).removeValue(value);
649                    case LEAF:
650                        if(leftIdx < 0) {
651                            return KEY_NOT_FOUND;
652                        } else {
653                            long oldPtr = ptrs[leftIdx];
654                            set(ArrayUtils.remove(keys, leftIdx), ArrayUtils.remove(ptrs, leftIdx));
655                            decrDataLength(value);
656                            return oldPtr;
657                        }
658                    default:
659                        throw new BTreeCorruptException("Invalid page type '" + ph.getStatus()
660                                + "' in removeValue");
661                }
662            }

664            /** @return pointer of matched items */
665            @Deprecated
666            private synchronized long[] removeValue(Value value, long pointer) throws IOException,
667                    DbException {
668                int leftIdx = searchLeftmostKey(keys, value, keys.length);
669                int rightIdx = isDuplicateAllowed() ? searchRightmostKey(keys, value, keys.length)
670                        : leftIdx;
671                switch(ph.getStatus()) {
672                    case BRANCH:
673                        leftIdx = (leftIdx < 0) ? -(leftIdx + 1) : leftIdx + 1;
674                        //FIXME keys may be separated nodes
675                        return getChildNode(leftIdx).removeValue(value, pointer);
676                    case LEAF:
677                        if(leftIdx < 0) {
678                            return new long[0];
679                        } else {
680                            int founds = 0;
681                            long[] matched = new long[rightIdx - leftIdx + 1];
682                            for(int i = leftIdx; i < rightIdx; i++) {
683                                long p = ptrs[i];
684                                if(p == pointer) {
685                                    set(ArrayUtils.remove(keys, i), ArrayUtils.remove(ptrs, i));
686                                    decrDataLength(value);
687                                    matched[founds++] = p;
688                                }
689                            }
690                            if(founds == 0) {
691                                return new long[0];
692                            }
693                            return (founds == matched.length) ? matched
694                                    : ArrayUtils.copyOfRange(matched, 0, founds);
695                        }
696                    default:
697                        throw new BTreeCorruptException("Invalid page type '" + ph.getStatus()
698                                + "' in removeValue");
699                }
700            }

717             * @see #write()
718             */
719            private boolean needSplit() {
720                int afterKeysLength = keys.length + 1;
721                if(afterKeysLength < LEAST_KEYS) {// at least 5 elements in a node
722                    return false;
723                }
724                if(afterKeysLength > Short.MAX_VALUE) {
725                    return true;
726                }
727                assert (prefix != null);
728                // CurrLength + one Long pointer + value length + one int (for value length)
729                // actual datalen is smaller than this datalen, because prefix is used.
730                int datalen = calculateDataLength();
731                int worksize = _fileHeader.getWorkSize();
732                return datalen > worksize;
733            }

735            /**
736             * Internal to the BTreeNode method
737             */
738            private void split() throws IOException, DbException {
739                final Value[] leftVals;
740                final Value[] rightVals;
741                final long[] leftPtrs;
742                final long[] rightPtrs;
743                final Value separator;
744    
745                final short vc = ph.getValueCount();
746                int pivot = vc / 2;
747    
748                // Split the node into two nodes
749                final byte pageType = ph.getStatus();
750                int leftLookup = 0;
751                switch(pageType) {
752                    case BRANCH:
753                        leftVals = new Value[pivot];
754                        leftPtrs = new long[leftVals.length + 1];
755                        rightVals = new Value[vc - (pivot + 1)];
756                        rightPtrs = new long[rightVals.length + 1];
757    
758                        System.arraycopy(keys, 0, leftVals, 0, leftVals.length);
759                        System.arraycopy(ptrs, 0, leftPtrs, 0, leftPtrs.length);
760                        System.arraycopy(keys, leftVals.length + 1, rightVals, 0, rightVals.length);
761                        System.arraycopy(ptrs, leftPtrs.length, rightPtrs, 0, rightPtrs.length);
762    
763                        separator = keys[leftVals.length];
764                        break;
765                    case LEAF:
766                        Value pivotLeft = keys[pivot - 1];
767                        Value pivotRight = keys[pivot];
768                        if(pivotLeft.equals(pivotRight)) {
769                            int leftmost = searchLeftmostKey(keys, pivotLeft, pivot - 1);
770                            int diff = pivot - leftmost;
771                            if(diff < 0 || diff > Short.MAX_VALUE) {
772                                throw new IllegalStateException("pivot: " + pivot + ", leftmost: "
773                                        + leftmost + "\nkeys: " + Arrays.toString(keys));
774                            }
775                            leftLookup = diff;
776                        }
777    
778                        leftVals = new Value[pivot];
779                        leftPtrs = new long[leftVals.length];
780                        rightVals = new Value[vc - pivot];
781                        rightPtrs = new long[rightVals.length];
782    
783                        System.arraycopy(keys, 0, leftVals, 0, leftVals.length);
784                        System.arraycopy(ptrs, 0, leftPtrs, 0, leftPtrs.length);
785                        System.arraycopy(keys, leftVals.length, rightVals, 0, rightVals.length);
786                        System.arraycopy(ptrs, leftPtrs.length, rightPtrs, 0, rightPtrs.length);
787    
788                        separator = getSeparator(leftVals[leftVals.length - 1], rightVals[0]);
789                        break;
790                    default:
791                        throw new BTreeCorruptException("Invalid page type in split: " + pageType);
792                }
793    
794                // Promote the pivot to the parent branch
795                if(parent == null) {
796                    // This can only happen if this is the root     
797                    final BTreeNode lNode = createBTreeNode(root, pageType, this);
798                    lNode.set(leftVals, leftPtrs);
799                    lNode.calculateDataLength();
800    
801                    final BTreeNode rNode = createBTreeNode(root, pageType, this);
802                    rNode.set(rightVals, rightPtrs);
803                    rNode.calculateDataLength();
804    
805                    if(pageType == LEAF) {
806                        setLeavesLinked(lNode, rNode);
807                    }
808    
809                    ph.setStatus(BRANCH);
810                    set(new Value[] { separator }, new long[] { lNode.page.getPageNum(),
811                            rNode.page.getPageNum() });
812                    calculateDataLength();
813    
814                } else {
815                    set(leftVals, leftPtrs);
816                    calculateDataLength();
817    
818                    final BTreeNode rNode = createBTreeNode(root, pageType, parent);
819                    rNode.set(rightVals, rightPtrs);
820                    rNode.calculateDataLength();
821    
822                    if(pageType == LEAF) {
823                        setLeavesLinked(this, rNode);
824                        if(leftLookup > 0) {
825                            rNode.ph.setLeftLookup((short) leftLookup);
826                        }
827                    }
828    
829                    final long leftPtr = page.getPageNum();
830                    parent.promoteValue(separator, rNode.page.getPageNum(), leftPtr);
831                }
832            }

834            /** Set leaves linked 
835             * @throws DbException */
836            private void setLeavesLinked(final BTreeNode left, final BTreeNode right)
837                    throws DbException {
838                final long leftPageNum = left.page.getPageNum();
839                final long rightPageNum = right.page.getPageNum();
840                final long origNext = left._next;
841                if(origNext != -1L) {
842                    right._next = origNext;
843                    BTreeNode origNextNode = getBTreeNode(root, origNext, parent);
844                    origNextNode._prev = rightPageNum;
845                    origNextNode.setDirty(true);
846                }
847                left._next = rightPageNum;
848                right._prev = leftPageNum;
849            }

851            private void promoteValue(final Value key, final long rightPointer, final long leftPtr)
852                    throws IOException, DbException {
853                final int leftIdx = searchRightmostKey(keys, key, keys.length);
854                int insertPoint = (leftIdx < 0) ? -(leftIdx + 1) : leftIdx + 1;
855                boolean found = false;
856                for(int i = insertPoint; i >= 0; i--) {
857                    final long ptr = ptrs[i];
858                    if(ptr == leftPtr) {
859                        insertPoint = i;
860                        found = true;
861                        break;
862                    } else {
863                        continue; // just for debugging
864                    }
865                }
866                if(!found) {
867                    throw new IllegalStateException("page#" + page.getPageNum() + ", insertPoint: "
868                            + insertPoint + ", leftPtr: " + leftPtr + ", ptrs: "
869                            + Arrays.toString(ptrs));
870                }
871                set(ArrayUtils.<Value> insert(keys, insertPoint, key), ArrayUtils.insert(ptrs, insertPoint + 1, rightPointer));
872                incrDataLength(key, rightPointer);
873    
874                // Check to see if we've exhausted the block
875                if(needSplit()) {
876                    split();
877                }
878            }

880            /** Gets shortest-possible separator for the pivot */
881            private Value getSeparator(final Value value1, final Value value2) {
882                int idx = value1.compareTo(value2);
883                if(idx == 0) {
884                    return value1.clone();
885                }
886                byte[] b = new byte[Math.abs(idx)];
887                value2.copyTo(b, 0, b.length);
888                return new Value(b);
889            }

891            /**
892             * Sets values and pointers.
893             * Internal (to the BTreeNode) method, not synchronized.
894             */
895            private void set(final Value[] values, final long[] ptrs) {
896                final int vlen = values.length;
897                if(vlen > Short.MAX_VALUE) {
898                    throw new IllegalArgumentException("entries exceeds limit: " + vlen);
899                }
900                this.keys = values;
901                this.ptrs = ptrs;
902                this.ph.setValueCount((short) vlen);
903                if(vlen > 1) {
904                    final int prevPreixLen = ph.getPrefixLength();
905                    this.prefix = getPrefix(values[0], values[vlen - 1]);
906                    final int prefixLen = prefix.getLength();
907                    assert (prefixLen <= Short.MAX_VALUE) : prefixLen;
908                    if(prefixLen != prevPreixLen) {
909                        final int diff = prefixLen - prevPreixLen;
910                        currentDataLen += diff;
911                        ph.setPrefixLength((short) prefixLen);
912                    }
913                } else {
914                    this.prefix = EmptyValue;
915                    ph.setPrefixLength((short) 0);
916                }
917                setDirty(true);
918                _cache.put(page.getPageNum(), this); // required? REVIEWME
919            }

925            private Value getPrefix(final Value v1, final Value v2) {
926                final int idx = Math.abs(v1.compareTo(v2)) - 1;
927                if(idx > 0) {
928                    final byte[] d2 = v2.getData();
929                    return new Value(d2, v2.getPosition(), idx);
930                } else {
931                    return EmptyValue;
932                }

935            /**
936             * Reads node only if it is not loaded yet
937             */
938            private synchronized void read() throws IOException, DbException {
939                if(!this.loaded) {
940                    Value v = readValue(page);
941                    DataInputStream in = new DataInputStream(v.getInputStream());
942                    // Read in the common prefix (if any)
943                    final short pfxLen = ph.getPrefixLength();
944                    final byte[] pfxBytes;
945                    if(pfxLen > 0) {
946                        pfxBytes = new byte[pfxLen];
947                        in.read(pfxBytes);
948                        this.prefix = new Value(pfxBytes);
949                    } else {
950                        pfxBytes = EmptyBytes;
951                        this.prefix = EmptyValue;
952                    }
953                    // Read in the Values
954                    Value prevKey = null;
955                    final int keyslen = ph.getValueCount();
956                    keys = new Value[keyslen];
957                    for(int i = 0; i < keyslen; i++) {
958                        final int valSize = in.readInt();
959                        if(valSize == -1) {
960                            prevKey.incrRefCount();
961                            keys[i] = prevKey;
962                        } else {
963                            byte[] b = new byte[pfxLen + valSize];
964                            if(pfxLen > 0) {
965                                System.arraycopy(pfxBytes, 0, b, 0, pfxLen);
966                            }
967                            if(valSize > 0) {
968                                in.read(b, pfxLen, valSize);
969                            }
970                            prevKey = new Value(b);
971                            keys[i] = prevKey;
972                        }
973                    }
974                    // Read in the pointers
975                    final int ptrslen = ph.getPointerCount();
976                    ptrs = new long[ptrslen];
977                    for(int i = 0; i < ptrslen; i++) {
978                        ptrs[i] = VariableByteCodec.decodeLong(in);
979                    }
980                    // Read in the links if current node is a leaf
981                    if(ph.getStatus() == LEAF) {
982                        this._prev = in.readLong();
983                        this._next = in.readLong();
984                    }
985                    this.currentDataLen = v.getLength();
986                    this.loaded = true;
987                }
988            }

990            private synchronized void write() throws IOException, DbException {
991                if(!dirty) {
992                    return;
993                }
994                if(LOG.isTraceEnabled()) {
995                    LOG.trace((ph.getStatus() == LEAF ? "Leaf " : "Branch ") + "Node#"
996                            + page.getPageNum() + " - " + Arrays.toString(keys));
997                }
998                final FastMultiByteArrayOutputStream bos = new FastMultiByteArrayOutputStream(_fileHeader.getWorkSize());
999                final DataOutputStream os = new DataOutputStream(bos);
1000    
1001                // write out the prefix
1002                final short prefixlen = ph.getPrefixLength();
1003                if(prefixlen > 0) {
1004                    prefix.writeTo(os);
1005                }
1006                // Write out the Values
1007                Value prevKey = null;
1008                for(int i = 0; i < keys.length; i++) {
1009                    final Value v = keys[i];
1010                    if(v == prevKey) {
1011                        os.writeInt(-1);
1012                    } else {
1013                        final int len = v.getLength();
1014                        final int size = len - prefixlen;
1015                        os.writeInt(size);
1016                        if(size > 0) {
1017                            v.writeTo(os, prefixlen, size);
1018                        }
1019                    }
1020                    prevKey = v;
1021                }
1022                // Write out the pointers
1023                for(int i = 0; i < ptrs.length; i++) {
1024                    VariableByteCodec.encodeLong(ptrs[i], os);
1025                }
1026                // Write out link if current node is a leaf
1027                if(ph.getStatus() == LEAF) {
1028                    os.writeLong(_prev);
1029                    os.writeLong(_next);
1030                }
1031    
1032                writeValue(page, new Value(bos.toByteArray()));
1033                setDirty(false);
1034            }

1036            private int calculateDataLength() {
1037                if(currentDataLen > 0) {
1038                    return currentDataLen;
1039                }
1040                final int vlen = keys.length;
1041                final short prefixlen = ph.getPrefixLength();
1042                int datalen = prefixlen + (vlen >>> 2) /* key size */;
1043                Value prevValue = null;
1044                for(int i = 0; i < vlen; i++) {
1045                    final long ptr = ptrs[i];
1046                    datalen += VariableByteCodec.requiredBytes(ptr);
1047                    final Value v = keys[i];
1048                    if(v == prevValue) {
1049                        continue;
1050                    }
1051                    final int keylen = v.getLength();
1052                    final int actkeylen = keylen - prefixlen; /* actual keys length */
1053                    datalen += actkeylen;
1054                    prevValue = v;
1055                }
1056                if(ph.getStatus() == LEAF) {
1057                    datalen += 16;
1058                }
1059                this.currentDataLen = datalen;
1060                return datalen;
1061            }

1063            private void incrDataLength(final Value value, final long ptr) {
1064                int datalen = currentDataLen;
1065                if(datalen == -1) {
1066                    datalen = calculateDataLength();
1067                }
1068                final int refcnt = value.incrRefCount();
1069                if(refcnt == 1) {
1070                    datalen += value.getLength();
1071                }
1072                datalen += VariableByteCodec.requiredBytes(ptr);
1073                datalen += 4 /* key size */;
1074                this.currentDataLen = datalen;
1075            }
1076    
1077            private void decrDataLength(final Value value) {
1078                int datalen = currentDataLen;
1079                final int refcnt = value.decrRefCount();
1080                if(refcnt == 0) {
1081                    datalen -= value.getLength();
1082                }
1083                datalen -= (4 + 8);
1084                this.currentDataLen = datalen;
1085            }

1087            /** find lest-most value which matches to the key */
1088            private synchronized long findValue(Value value) throws DbException {
1089                if(value == null) {
1090                    throw new DbException("Can't search on null Value");
1091                }
1092                int idx = searchLeftmostKey(keys, value, keys.length);
1093                switch(ph.getStatus()) {
1094                    case BRANCH:
1095                        idx = idx < 0 ? -(idx + 1) : idx + 1;
1096                        return getChildNode(idx).findValue(value);
1097                    case LEAF:
1098                        if(idx < 0) {
1099                            return KEY_NOT_FOUND;
1100                        } else {
1101                            int lookup = ph.getLeftLookup();
1102                            if(lookup > 0) {
1103                                BTreeNode leftmostNode = this;
1104                                while(true) {
1105                                    final BTreeNode prevNode = getBTreeNode(root, leftmostNode._prev, parent);
1106                                    final int prevLookup = prevNode.ph.getLeftLookup();
1107                                    if(prevLookup == 0) {
1108                                        break;
1109                                    }
1110                                    leftmostNode = prevNode;
1111                                    lookup = prevLookup;
1112                                }
1113                                final long[] leftmostPtrs = leftmostNode.ptrs;
1114                                final int offset = leftmostPtrs.length - lookup;
1115                                return leftmostPtrs[offset];
1116                            } else {
1117                                return ptrs[idx];
1118                            }
1119                        }
1120                    default:
1121                        throw new BTreeCorruptException("Invalid page type '" + ph.getStatus()
1122                                + "' in findValue");
1123                }
1124            }

1126            /** 
1127             * Scan the leaf node. 
1128             * Note that keys might be shortest-possible value.
1129             */
1130            private void scanLeaf(IndexQuery query, BTreeCallback callback, boolean edge) {
1131                assert (ph.getStatus() == LEAF) : LEAF;
1132                Value[] conds = query.getOperands();
1133                switch(query.getOperator()) {
1134                    case BasicIndexQuery.EQ: {
1135                        if(!edge) {
1136                            for(int i = 0; i < keys.length; i++) {
1137                                callback.indexInfo(keys[i], ptrs[i]);
1138                            }
1139                            return;
1140                        }
1141                        final int leftIdx = searchLeftmostKey(keys, conds[0], keys.length);
1142                        if(leftIdx >= 0) {
1143                            assert (isDuplicateAllowed());
1144                            final int rightIdx = searchRightmostKey(keys, conds[conds.length - 1], keys.length);
1145                            for(int i = leftIdx; i <= rightIdx; i++) {
1146                                callback.indexInfo(keys[i], ptrs[i]);
1147                            }
1148                        }
1149                        break;
1150                    }
1151                    case BasicIndexQuery.NE: {
1152                        int leftIdx = searchLeftmostKey(keys, conds[0], keys.length);
1153                        int rightIdx = isDuplicateAllowed() ? searchRightmostKey(keys, conds[conds.length - 1], keys.length)
1154                                : leftIdx;
1155                        for(int i = 0; i < ptrs.length; i++) {
1156                            if(i < leftIdx || i > rightIdx) {
1157                                callback.indexInfo(keys[i], ptrs[i]);
1158                            }
1159                        }
1160                        break;
1161                    }
1162                    case BasicIndexQuery.BWX:
1163                    case BasicIndexQuery.BW:
1164                    case BasicIndexQuery.START_WITH:
1165                    case BasicIndexQuery.IN: {
1166                        if(!edge) {
1167                            for(int i = 0; i < keys.length; i++) {
1168                                if(query.testValue(keys[i])) {
1169                                    callback.indexInfo(keys[i], ptrs[i]);
1170                                }
1171                            }
1172                            return;
1173                        }
1174                        int leftIdx = searchLeftmostKey(keys, conds[0], keys.length);
1175                        if(leftIdx < 0) {
1176                            leftIdx = -(leftIdx + 1);
1177                        }
1178                        int rightIdx = searchRightmostKey(keys, conds[conds.length - 1], keys.length);
1179                        if(rightIdx < 0) {
1180                            rightIdx = -(rightIdx + 1);
1181                        }
1182                        for(int i = leftIdx; i < ptrs.length; i++) {
1183                            if(i <= rightIdx && query.testValue(keys[i])) {
1184                                callback.indexInfo(keys[i], ptrs[i]);
1185                            }
1186                        }
1187                        break;
1188                    }
1189                    case BasicIndexQuery.NBWX:
1190                    case BasicIndexQuery.NBW:
1191                    case BasicIndexQuery.NOT_START_WITH: {
1192                        int leftIdx = searchLeftmostKey(keys, conds[0], keys.length);
1193                        if(leftIdx < 0) {
1194                            leftIdx = -(leftIdx + 1);
1195                        }
1196                        int rightIdx = searchRightmostKey(keys, conds[conds.length - 1], keys.length);
1197                        if(rightIdx < 0) {
1198                            rightIdx = -(rightIdx + 1);
1199                        }
1200                        for(int i = 0; i < ptrs.length; i++) {
1201                            if((i <= leftIdx || i >= rightIdx) && query.testValue(keys[i])) {
1202                                callback.indexInfo(keys[i], ptrs[i]);
1203                            }
1204                        }
1205                        break;
1206                    }
1207                    case BasicIndexQuery.LT: {
1208                        int leftIdx = searchLeftmostKey(keys, conds[0], keys.length);
1209                        if(leftIdx < 0) {
1210                            leftIdx = -(leftIdx + 1); // insertion point
1211                        }
1212                        for(int i = 0; i < leftIdx; i++) {
1213                            callback.indexInfo(keys[i], ptrs[i]);
1214                        }
1215                        break;
1216                    }
1217                    case BasicIndexQuery.LE: {
1218                        int leftIdx = searchRightmostKey(keys, conds[0], keys.length);
1219                        if(leftIdx < 0) {
1220                            leftIdx = -(leftIdx + 1); // insertion point
1221                        }
1222                        if(leftIdx >= ptrs.length) {
1223                            leftIdx = ptrs.length - 1;
1224                        }
1225                        for(int i = 0; i <= leftIdx; i++) {
1226                            callback.indexInfo(keys[i], ptrs[i]);
1227                        }
1228                        break;
1229                    }
1230                    case BasicIndexQuery.GT: {
1231                        int rightIdx = searchRightmostKey(keys, conds[0], keys.length);
1232                        if(rightIdx < 0) {
1233                            rightIdx = -(rightIdx + 1);
1234                        }
1235                        for(int i = rightIdx + 1; i < ptrs.length; i++) {
1236                            callback.indexInfo(keys[i], ptrs[i]);
1237                        }
1238                        break;
1239                    }
1240                    case BasicIndexQuery.GE: {
1241                        int rightIdx = searchLeftmostKey(keys, conds[0], keys.length);
1242                        if(rightIdx < 0) {
1243                            rightIdx = -(rightIdx + 1);
1244                        }
1245                        for(int i = rightIdx; i < ptrs.length; i++) {
1246                            callback.indexInfo(keys[i], ptrs[i]);
1247                        }
1248                        break;
1249                    }
1250                    case BasicIndexQuery.ANY:
1251                        for(int i = 0; i < ptrs.length; i++) {
1252                            callback.indexInfo(keys[i], ptrs[i]);
1253                        }
1254                        break;
1255                    case BasicIndexQuery.NOT_IN:
1256                    default:
1257                        for(int i = 0; i < ptrs.length; i++) {
1258                            if(query.testValue(keys[i])) {
1259                                callback.indexInfo(keys[i], ptrs[i]);
1260                            }
1261                        }
1262                        break;
1263                }
1264            }

1266            private BTreeNode getLeafNode(SearchType searchType, Value key) throws IOException,
1267                    DbException {
1268                final byte nodeType = ph.getStatus();
1269                switch(nodeType) {
1270                    case BRANCH:
1271                        switch(searchType) {
1272                            case LEFT: {
1273                                int leftIdx = searchLeftmostKey(keys, key, keys.length);
1274                                leftIdx = leftIdx < 0 ? -(leftIdx + 1) : leftIdx + 1;
1275                                return getChildNode(leftIdx).getLeafNode(searchType, key);
1276                            }
1277                            case RIGHT: {
1278                                int rightIdx = searchRightmostKey(keys, key, keys.length);
1279                                rightIdx = rightIdx < 0 ? -(rightIdx + 1) : rightIdx + 1;
1280                                return getChildNode(rightIdx).getLeafNode(searchType, key);
1281                            }
1282                            case LEFT_MOST:
1283                                return getChildNode(0).getLeafNode(searchType, key);
1284                            case RIGHT_MOST:
1285                                int rightIdx = ptrs.length - 1;
1286                                assert (rightIdx >= 0);
1287                                return getChildNode(rightIdx).getLeafNode(searchType, key);
1288                            default:
1289                                throw new IllegalStateException();
1290                        }
1291                    case LEAF:
1292                        switch(searchType) {
1293                            case LEFT: {
1294                                BTreeNode leftmostNode = this;
1295                                if(keys[0].equals(key)) {
1296                                    int lookup = ph.getLeftLookup();
1297                                    while(lookup > 0) {
1298                                        leftmostNode = getBTreeNode(root, leftmostNode._prev, parent);
1299                                        int keylen = leftmostNode.keys.length;
1300                                        if(lookup < keylen) {
1301                                            break;
1302                                        }
1303                                        lookup = leftmostNode.ph.getLeftLookup();
1304                                        if(lookup == 0) {
1305                                            break;
1306                                        }
1307                                        Value firstKey = leftmostNode.keys[0];
1308                                        if(!firstKey.equals(key)) {
1309                                            break;
1310                                        }
1311                                    }
1312                                }
1313                                return leftmostNode;
1314                            }
1315                            case RIGHT_MOST:
1316                                if(_next != -1L) {
1317                                    BTreeNode nextNode = getBTreeNode(root, _next, parent);
1318                                    throw new IllegalStateException("next=" + _next + ".. more leaf ["
1319                                            + nextNode + "] exists on the right side of leaf ["
1320                                            + this.toString() + "]\n parent-ptrs: "
1321                                            + Arrays.toString(parent.ptrs));
1322                                }
1323                                break;
1324                            case LEFT_MOST:
1325                                if(_prev != -1L) {
1326                                    BTreeNode prevNode = getBTreeNode(root, _prev, parent);
1327                                    throw new IllegalStateException("prev=" + _prev + ".. more leaf ["
1328                                            + prevNode + "] exists on the left side of leaf ["
1329                                            + this.toString() + "]\n parent-ptrs: "
1330                                            + Arrays.toString(parent.ptrs));
1331                                }
1332                                break;
1333                            default:
1334                                break;
1335                        }
1336                        return this;
1337                    default:
1338                        throw new BTreeCorruptException("Invalid page type in query: " + nodeType);
1339                }
1340            }

1342            @Override
1343            public String toString() {
1344                final StringBuilder buf = new StringBuilder();
1345                final long rootPage = root.getPage();
1346                BTreeNode pn = this;
1347                while(true) {
1348                    final long curPageNum = pn.page.getPageNum();
1349                    buf.append(curPageNum);
1350                    pn = pn.parent;
1351                    if(pn == null) {
1352                        if(curPageNum != rootPage) {
1353                            buf.append("<-?");
1354                        }
1355                        break;
1356                    } else {
1357                        buf.append("<-");
1358                    }
1359                }
1360                return buf.toString();
1361            }
1362        }
